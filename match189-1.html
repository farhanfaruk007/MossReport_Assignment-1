import heapq

class graph_object:
  def __init__(self):
    self.graph = {}

  def add_edge(self,node,neighbor,cost = 0):
    if node not in self.graph:
      self.graph[node] = []
    if neighbor not in self.graph:
      self.graph[neighbor] = []
    if (node,cost) not in self.graph[neighbor]:
      self.graph[neighbor].append((node,cost))
    if (neighbor,cost) not in self.graph[node]:
      self.graph[node].append((neighbor,cost))

  def get_neighbors(self,node):
    if node in self.graph:
      return self.graph[node]
    else:
      return "<!Node not Found!>"



  def get_graph(self):
    return self.graph

def heuristic(nod,lis) -> int :
  for n,val in lis:
    if n == nod:
      return val

  return 0





def a_star_search(graph_object,initial_state,goal,lis):
  visited = set()
  unvisited = [(0,initial_state)]
  path_cost = {initial_state:0}
  parents = {}

  while unvisited:
    _, current = heapq.heappop(unvisited)

    if current == goal:
      path = []
      total_cost = path_cost[current]
      while current in parents:
        path.append(current)
        current = parents[current]
      path.append(initial_state)
      path.reverse()
      return path,total_cost

    visited.add(current)
    for neighbor,cost in graph_object.get_neighbors(current):
      new_cost = path_cost[current] + cost
      if new_cost < path_cost.get(neighbor, float('inf')):
        parents[neighbor] = current
        path_cost[neighbor] = new_cost
        priority = new_cost + heuristic(neighbor,lis)
        heapq.heappush(unvisited,(priority,neighbor))

  return None,0
