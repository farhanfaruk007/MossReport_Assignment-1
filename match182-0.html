import heapq

graph = {}
heuristic = {}


inp = open('Input file.txt', 'r')
for i in inp:
    line = i.strip().split(' ')
    graph[line[0]] = []
    heuristic[line[0]] = int(line[1])
    len = 0
    for l in line:
        len += 1
    for j in range(2, len, 2):
        graph[line[0]].append((line[j], int(line[j+1])))
#print(graph)
#print(heuristic)

start = input('Start: ')
des = input('Destination: ')

f = [(start,heuristic[start])]
path_cost = {start : 0}
visited = []
v_city = []
parent = {(start,heuristic[start]) : None}
heapq.heapify(f)
node = ( '',0)


while node[0] != des:
      node = heapq.heappop(f)
      visited.append(node)
      v_city.append(node[0])
      for i in (graph[node[0]]):
          if i[0] not in v_city:
              cost = path_cost[node[0]] + i[1]
              path_cost[i[0]] = cost
              total_cost = heuristic[i[0]] + cost
              heapq.heappush(f, (i[0],total_cost))
              parent[(i[0],total_cost)] = node
#print(path_cost)
#print(parent)
#print(visited)
#print(v_city)

path = [des]
len2 = 0
for i in visited:
  len2+=1
len2=len2-1
#print(len2)
#print(visited[len2])
current = visited[len2]
while current != (start,heuristic[start]):
    current = parent[current]
    if current in visited:
      path.append(current[0])
path.reverse()
outputfile = open('Output file.txt', 'w')
len3=0
for i in path:
  len3+=1
P = 'Path: '
for i in range (len3):
    if i != len3 - 1:
      P+= f"{path[i]} -> "
    else:
      P += path[i] + '\n'
#print(str1)
outputfile.write(P)
total_dis=f"Total distance: {visited[len2][1] - heuristic[visited[len2][0]]} km"
#print(total_dis)
outputfile.write(total_dis)
