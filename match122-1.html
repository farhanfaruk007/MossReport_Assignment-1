def adj(inp):
    h = {}
    paths = {}
    for i in range(20):
        line = list((inp.readline().split()))
        for e in range(len(line)):
            if e == 1:
                h[line[e - 1]] = int(line[e])
                paths[line[e - 1]] = []

            else:
                if e % 2 == 1:
                    paths[line[0]].append((line[e - 1], int(line[e])))

    return paths, h




class ASTAR:
    def __init__(self, paths, h):
        self.paths = paths
        self.heuristic = h

    def heu(self, node):
        return self.heuristic[node]

    def neighbouring_node(self, n):
        return self.paths[n]

    def searching(self, st, goal):

        dist = {}
        parentsList = {}
        finalPaths = []

        dist[st] = 0
        parentsList[st] = st

        openedList = set([st])
        closedList = set([])

        while openedList:


            v = None
            for e in openedList:
                if v == None or dist[e] + self.heu(e) < dist[v] + self.heu(v):
                    v = e

            if v == None:
                print('There is no path!')

            if v == goal:
                while parentsList[v] != v:
                    finalPaths.append(v)
                    v = parentsList[v]

                finalPaths.append(st)
                finalPaths.reverse()

                total = 0
                a = 0

                while a < (len(finalPaths) - 1):
                    for b, c in self.paths.items():
                        if b == finalPaths[a]:
                            for x, y in c:
                                if x == finalPaths[a + 1]:
                                    total += y
                                    a += 1
                                    break
                            break

                path_str = "Paths: "
                for i in finalPaths:
                    if i != finalPaths[-1]:
                        path_str = path_str + i + " -> "
                    else:
                        path_str = path_str + i

                out.write(f"{path_str}\nTotal distance: {total} km\n\n")
                return finalPaths



            for (n, cost) in self.neighbouring_node(v):
                if n not in openedList and n not in closedList:
                    openedList.add(n)
                    parentsList[n] = v
                    dist[n] = dist[v] + cost
                else:
                    if dist[n] > dist[v] + cost:


                        dist[n] = dist[v] + cost
                        parentsList[n] = v
                        if n in closedList:
                            closedList.remove(n)
                            openedList.add(n)

            openedList.remove(v)
            closedList.add(v)

        out.write('There is no path!')
