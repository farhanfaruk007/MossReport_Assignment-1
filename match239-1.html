import heapq

class CREATION:
    def __init__(self):
        self.created_list = { }

    def edges_ADD(self, aNode, bNode, STORE=0):

        if bNode not in   self.created_list:
            self.created_list[bNode] = []
        if aNode not in self.created_list:
            self.created_list[aNode] = []

        if (aNode, STORE) not in  self.created_list[bNode]:
            self.created_list[bNode].append((aNode,STORE))

        if (bNode, STORE) not in self.created_list[aNode]:
            self.created_list[aNode].append((bNode, STORE))

    def PICTURE(self):

        return self.created_list


    def NEIGHBR(self, i):

        return     self.created_list[i] if i  in self.created_list else [ ]



def WEIGHT (i, WEIGHTED_HEURIS) -> int:

    for n, W in  WEIGHTED_HEURIS:
        if n == i:
            return W
    return 0

def aSTRsear(d, start, goal,  WEIGHTED_HEURI):
    v = set()
    nON_VIS = [(0, start)]
    MM = {}
    value = {start: 0}


    while nON_VIS:

        _,REC = heapq.heappop(nON_VIS)

        if REC == goal:

            ROAD = [ ]
            TOTAL = value[REC]
            while  REC in MM:

                REC = MM[REC]
                ROAD.append(REC)
            ROAD.append(start)

            return list(reversed(ROAD)), TOTAL

        v.add(REC)


        for neighbor, neighborCost in d.NEIGHBR(REC):
            tentative_cost = value[REC] + neighborCost

            if neighbor in v or tentative_cost >= value.get(neighbor, float('inf')):
                continue

            if tentative_cost < value.get(neighbor, float('inf')):
                MM[neighbor] = REC
                value[neighbor] = tentative_cost

                totalCost = tentative_cost + WEIGHT(neighbor,  WEIGHTED_HEURIS)
                heapq.heappush(nON_VIS, (totalCost, neighbor))

    return None, 0  # No path found
