from queue import PriorityQueue

def read_graph_and_heuristics(fileN):
    g = {}
    h = {}

    with open(fileN, 'r') as f:
        for row in f:
            place_n_val = row.split()
            place = place_n_val[0]
            h_value = int(place_n_val[1])
            h[place] = h_value
            children = place_n_val[2:]
            g[place] = []

            for i in range(0, len(children), 2):
                n = children[i]
                dis = int(children[i+1])
                g[place].append((n, dis))
    return g, h

def a_star_search(g, h, start, end):
    bag = PriorityQueue()
    bag.put((0, start))
    dict_cost = {start: 0}
    visit = {start: None}

    while not bag.empty():
        new, node = bag.get()
        if node == end:
            p = []
            while node:
                p.append(node)
                node = visit[node]
            return p[::-1], dict_cost[end]

        for n, dis in g[node]:
            new_cost = dict_cost[node] + dis
            if n not in dict_cost or dict_cost[n] > new_cost:
                dict_cost[n] = new_cost
                f_x = new_cost + h[n]
                bag.put((f_x, n))
                visit[n] = node
            else:
                pass

    return None, None



def main():
    fileN = "input1_file.txt"
    g, h = read_graph_and_heuristics(fileN)
    start = input("input the start node: ")
    end = input("input the destination: ")
    path, distance = a_star_search(g, h, start, end)

    if path:
        print("Path:", path)
        print("Total distance:", distance)
    else:
        print("No path found.")

if __name__ == "__main__":
    main()
