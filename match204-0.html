import matplotlib.pyplot as plt

def load_map(txt_file):
    heuristics = {}
    path = {} 
    
    with open(txt_file, 'r') as file:
        for l1 in file:
            elem = l1.split()
            cty = elem[0]
            h_weight = int(elem[1])
            heuristics[cty] = h_weight
            child_lst = elem[2:]
            path[cty] = {}

            for i in range(0, len(child_lst), 2):


                n1 = child_lst[i]
                dist = int(child_lst[i + 1])
                path[cty][n1] = dist

    return path, heuristics




def a_star(path, heuristics, start, end):
    open_lst = {start}
    parent_lst = {}

    lowest_p_cost = {}
    total_ecost = {}
    limit = 999999999999

    for cty in path:
      lowest_p_cost[cty] = limit
      total_ecost[cty] = limit


    lowest_p_cost[start] = 0
    total_ecost[start] = heuristics[start]

    while open_lst:
        curr1 = None
        min_cost = limit

        for cty in open_lst:
            if total_ecost[cty] < min_cost:
                min_cost = total_ecost[cty]
                curr1 = cty

        if curr1 == end:
            return reconst(parent_lst, curr1), lowest_p_cost[end]



        open_lst.remove(curr1)

        for n1, dist in path[curr1].items():
            new_cost = lowest_p_cost[curr1] + dist

            if new_cost < lowest_p_cost[n1]:
                parent_lst[n1] = curr1
                lowest_p_cost[n1] = new_cost
                total_ecost[n1] = new_cost + heuristics[n1]

                if n1 not in open_lst:
                    open_lst.add(n1)

    return None, None





def reconst(parent_lst, curr1):
    path = [curr1]
    while curr1 in parent_lst:
        curr1 = parent_lst[curr1]
        path.append(curr1)
    return path[::-1]



start = 'Arad'
end = 'Bucharest'
path, heuristics = load_map('input.txt')
path, total_dist = a_star(path, heuristics, start, end)


if path:
    print("Path:", " -> ".join(path))
    print("Total distance:", total_dist, "km")
    
else:
    print("No path found.")
