import heapq
def read_data(lines):
  graph={}
  for line in lines:
    parts=line.split()
    city = parts[0]
    heuristic = int(parts[1])
    neighbors = {}
    for i in range(2, len(parts), 2):
            neighbor_city = parts[i]


            travel_distance = int(parts[i + 1])
            neighbors[neighbor_city] = travel_distance
    graph[city] = {"Heuristic": heuristic, "Neighbors": neighbors}
  return graph
def A_star(graph, start,destination):
  open_list = [(0, start)]
  g_n = {city: float("inf") for city in graph}
  g_n[start] = 0
  previous_city = {city: None for city in graph}
  visited_cities = set()
  while open_list:
    current_cost, current_city = heapq.heappop(open_list)
    visited_cities.add(current_city)


    if current_city == destination:
            path = []
            while current_city is not None:
                path.append(current_city)
                current_city = previous_city[current_city]
            return path[::-1], g_n[destination]

    for neighbor, distance in graph[current_city]["Neighbors"].items():
            if neighbor in visited_cities:
                continue
            temp_cost = g_n[current_city] + distance


            if temp_cost < g_n[neighbor]:
                previous_city[neighbor] = current_city
                g_n[neighbor] = temp_cost

                f_cost = temp_cost + graph[neighbor]["Heuristic"]
                heapq.heappush(open_list, (f_cost, neighbor))


  return None

with open("/content/input.txt") as f:
    lines=f.readlines()

    graph=read_data (lines)
start= input("Start Node: ")
destination = input("Destination: ")
if start not in graph or destination not in graph:
    print("The start or destination city is not found in the graph.")
else:
    result = A_star(graph, start, destination)
    if result:
      path, total_distance = result
      print("Path:", " -> ".join(path))
      print("Total distance:", total_distance, "km")
    else:
        print("No path found.")
