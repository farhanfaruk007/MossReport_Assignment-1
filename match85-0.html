# A_star algorithm
import heapq
inp_f = open('/content/input.txt')


out_f = open('/content/output.txt', 'w')
dict_map = {} # graph representation
heuristic_val = {}
list_m = inp_f.readlines()  # inputs of graph

for i in list_m:
    list_node = i.strip().split()
    parent = list_node[0]                         # parent nodes
    heu_val = int(list_node[1])            #parent node's heuristic value
    heuristic_val[parent] = heu_val         # storing heuristic value
    dict_map[list_node[0]] = {}                 # current parent's children
    for j in range(2, len(list_node), 2):
        dict_map[list_node[0]][list_node[j]] = int(list_node[j + 1])      #creating graph



def A_star(start, goal, direct_line, graph):
    queue = []
    map = {}
    h_val = direct_line # heuristic value
    way = {}
    route = {}
    heapq.heappush(queue, (0, start))
    way[start] = None
    route[start] = 0
    flag = 1
    while flag == 1:
        pre_par = heapq.heappop(queue)[1] # current parent
        if pre_par != goal:
            for child in graph[pre_par]:
                g_val = route[pre_par] + graph[pre_par][child]
                if child not in route or g_val < route[child]:
                    route[child] = g_val
                    f_val = g_val + h_val[child]
                    heapq.heappush(queue, (f_val, child))
                    way[child] = pre_par
        else:
            flag = 0
    return way, route
#Taking inputs for start and end destination


Begin = input("Enter start destination: ")
goal = input("Enter end destination: ")
way, route = A_star(Begin, goal, heuristic_val, dict_map)
ans_way = []   # founded route or answer
ans_way.append(goal) # for printing in backtrack
temp = goal

#Showing output
while temp != Begin:
    ans_way.append(way[temp])
    temp = way[temp]
n_val = len(ans_way)
if len(ans_way) == 0:
    print("NO PATH FOUND", file = out_f)
else:


    ans = "Path: "
    for i in range(n_val - 1, -1, -1):
        if ans_way[i] == goal:
            ans += ans_way[i]
        else:
            ans += ans_way[i] + " --> "
    print(ans, file = out_f)
    print("Total distance: ", route[goal], "km", file = out_f)

inp_f.close()
out_f.close()
