import heapdict



h_t = {}
g = {}
with open("Input.txt") as inp:
    for l in inp:
        l = l.strip().split(" ")
        h_t[l[0]] = int(l[1])
        temp = {}
        for i in range(2, len(l), 2):
            temp[l[i]] = int(l[i + 1])
        g[l[0]] = temp

def a_star(start, dest, g, h_t):
    
    nodes = list(h_t.keys())
    que = heapdict.heapdict()
    par_tracker = [None] * len(h_t)
    par_cost = [float("inf")] * len(h_t)

    start_ind = nodes.index(start)


    que[start] = h_t[start]
    par_cost[start_ind] = 0

    while len(que) != 0:
        n = que.popitem()
        if n[0] == dest:
            cur = dest
            path = [dest]
            cur_ind = nodes.index(cur)
            destination_ind = nodes.index(dest)

            while par_tracker[cur_ind] != None:
                path.insert(0, par_tracker[cur_ind])
                cur = par_tracker[cur_ind]
                cur_ind = nodes.index(cur)

            print("Path: " + " -> ".join(path))
            print("Total distance:", par_cost[destination_ind], "km")
            return 0

        else: 
            for child in g[n[0]]:
                par_ind = nodes.index(n[0])
                child_ind = nodes.index(child)
                n_cost = g[n[0]][child]


                g_cost = par_cost[par_ind] + n_cost

                if g_cost < par_cost[child_ind]:
                    par_cost[child_ind] = g_cost
                    par_tracker[child_ind] = n[0]
                    f_n = g_cost + h_t[child]
                    que[child] = f_n

    print("NO PATH FOUND")
    return 1

start = input("Start node: ").capitalize()
dest = input("Destination: ").capitalize()

a_star(start, dest, g, h_t)
