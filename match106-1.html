import heapq


input = open("input.txt", "r")

def Heuristics(input):
    heu_dict = {}
    for i in input:


        Heu_val = i.strip().split()
        heu_dict[Heu_val[0]]=int(Heu_val[1])
    return heu_dict


input.seek(0)



def Graph(input):
    adj_list = {}
    for j in input:
        val = j.strip().split()
        current_nodes = val[0]

        for k in range(0,len(val[2:]),2):
            neighbour = val[2:][k]
            cost = int(val[2:][k+1])
            if current_nodes not in adj_list:
                adj_list[current_nodes] = {}
            adj_list[current_nodes][neighbour] = cost
    return adj_list


heuristic = Heuristics(input)

input.seek(0)
graph = Graph(input)






def Astar(graph, start, goal, heuristics):
  notVisited = [(0, start)]
  Visited = set()
  parent = {}
  cost = {}
  parent[start] = None
  cost[start] = 0


  while notVisited:


    n, current = heapq.heappop(notVisited)
    if current in Visited:
      continue
    Visited.add(current)

    if current == goal:
      path=[]
      total_cost = cost[current]
      while current is not None:
        path.append(current)
        current = parent[current]
      path.reverse()
      return path, total_cost

    for neighbour, path_cost in graph.get(current, {}).items():
      new_cost = cost[current] + path_cost
      if neighbour not in cost or new_cost < cost[neighbour]:
        cost[neighbour] = new_cost
        priority = new_cost + heuristics.get(neighbour, float('inf'))
        heapq.heappush(notVisited, (priority, neighbour))
        parent[neighbour] = current
  return None, float('inf')

path, total_cost = Astar(graph, 'Arad', 'Bucharest', heuristic)
print("Optimal path:", path)
print("Total_Cost:", total_cost)
