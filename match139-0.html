import math

from queue import PriorityQueue

f1 = open("Input1.txt", "r")
r = f1.read().split("\n")
graph = {}
h_v = {}                 #For storing heuristic values
#print(r)

for i in r:
    #print(i)
    list = i.split(" ")
    #print(list)
    h_v[list[0]] = int(list[1])             #assigning key-value into the h_v dict
    #print(h_v)
    d = []

    for x in range(2,len(list)-1,2):
        #print(list[x])
        if x+1 < len(list):
          d.append((list[x], int(list[x + 1])))
    #print(graph)
    graph[list[0]] = d


s_n =input("Start Node: ")
des_n =input("Destination Node: ")
def Astar_search(graph, start, dest, heu_val):
    queue = PriorityQueue()
    p = {}
    dis = {}
    total_cost = {}


    for n in graph.keys():
        dis[n] = math.inf
        total_cost[n] = math.inf
        p[n] = None


    

    dis[start] = 0

    total_cost[start] = heu_val[start]
    queue.put((total_cost[start], start))
    while not queue.empty():
        curr_cost, curr_node = queue.get()


        if curr_node == dest:
            path = []
            while curr_node is not None:


                path.append(curr_node)
                curr_node = p[curr_node]
            path.reverse()
            return path, dis[dest]


        for nbr, weight in graph[curr_node]:
            new_dis = dis[curr_node] + weight
            total_cost_beside = new_dis + heu_val[ nbr]

            if new_dis < dis[ nbr]:
                dis[ nbr] = new_dis
                p[ nbr] = curr_node
                queue.put((total_cost_beside, nbr))

    return "NO PATH FOUND", 0

pathway, cost = Astar_search(graph, s_n, des_n, h_v)
if cost != 0:
    print("Path:", ' -> '.join(pathway))
    print(f'Total distance: {cost} km')
else:
    print(pathway)
