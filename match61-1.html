"""A* search function

"""



def A_star(start, goal, graph, hn):
    path_list = []
    fn = 0 + hn[start]
    gn = 0
    path = [start]
    path_list = [(start, fn, gn, path)]

    for node in graph:
        graph[node] = {'visited': False, 'neighbors': graph[node]}

    while path_list:
        for i in range(len(path_list)):
            for j in range(0, len(path_list) - i - 1):
                if path_list[j][1] > path_list[j + 1][1]:
                    path_list[j], path_list[j + 1] = path_list[j + 1], path_list[j]

        node, fn, gn, path = path_list.pop(0)

        if graph[node]['visited']:
            continue

        graph[node]['visited'] = True

        if node == goal:
            return path, gn

        for adj_node, g_n in graph[node]['neighbors']:
            if not graph[adj_node]['visited']:
                path_list.append((adj_node, gn + g_n + hn[adj_node], gn + g_n, path + [adj_node]))

    return None, float('inf')
