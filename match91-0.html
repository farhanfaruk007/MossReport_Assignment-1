import heapq
inp = open('/content/sample_data/Input file.txt',"r")
g_map = {}
h_val = {}




for a in inp :
    t = a.split(' ')
    h_val[t[0]] = int(t[1])
    g_map[t[0]] = []

    for b in range(2, len(t)-1, 2):
        city = t[b]
        dist = t[b + 1]
        g_map[t[0]].append((city, int(dist)))


start = input('Please Enter Start point as String: ')
goal = input('Please Enter Destination city as String: ')


def Optimal_Path(parent_node, start, goal, dist):
    p = []
    p.append(goal)
    t = goal

    while t != None:
        p.append(parent_node[t])
        t = parent_node[t]

    p.reverse()

    if p == None:
        with open('Output file.txt', "w") as f:
            f.write('NO PATH FOUND.\n')
    else:
        with open('Output file.txt', "w") as f:
            f.write(f'Path: {" -> ".join(p[1:])}\n')
            f.write(f'Total distance: {dist[goal]} km\n')

def A_star(start, goal):

    parent_node = {start : None}
    dist = {start: 0}
    q = []  # queue



    heapq.heappush(q, (h_val[start], start))


    while q:
        t = heapq.heappop(q)
        for a in g_map[t[1]] :
            t_d = a[1] + dist[t[1]]



            if (a[0] not in dist) or (t_d < dist[a[0]]):
                dist[a[0]] = t_d
                path_cost = dist[a[0]] + h_val[a[0]]
                heapq.heappush(q, (path_cost, a[0]))
                parent_node[a[0]] = t[1]



    Optimal_Path(parent_node, start, goal, dist)
A_star(start, goal)
