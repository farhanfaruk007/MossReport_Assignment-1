import heapq



class Graph:
    def __init__(self):
        self.edges = {}
        self.h = {}

    def add_edge(self, from_node, to_node, distance):
        if from_node not in self.edges:
            self.edges[from_node] = []


        self.edges[from_node].append((to_node, distance))

    def set_heuristic(self, node, h_value):
        self.h[node] = h_value

    def get_neighbors(self, node):
        return self.edges.get(node, [])

    def heuristic(self, node):
        return self.h.get(node, float('inf'))

def a_star_search(graph, start, end):
    open_list = []
    heapq.heappush(open_list, (0 + graph.heuristic(start), 0, start, []))
    closed_list = set()

    while open_list:
        _, cost, current_node, path = heapq.heappop(open_list)

        if current_node in closed_list:
            continue

        path = path + [current_node]

        if current_node == end:
            return path, cost

        closed_list.add(current_node)

        for neighbor, distance in graph.get_neighbors(current_node):
            if neighbor not in closed_list:
                heuristic_cost = cost + distance + graph.heuristic(neighbor)
                heapq.heappush(open_list, (heuristic_cost, cost + distance, neighbor, path))

    return None, float('inf')

def read_input(file_path):
    graph = Graph()
    with open(file_path, 'r') as file:
        lines = file.readlines()
        for line in lines:


            parts = line.strip().split()
            node = parts[0]
            heuristic = int(parts[1])
            graph.set_heuristic(node, heuristic)
            for i in range(2, len(parts), 2):
                neighbor = parts[i]
                distance = int(parts[i+1])
                graph.add_edge(node, neighbor, distance)
    return graph

file_path = 'input.txt'
graph = read_input(file_path)

start = input("Start node: ").strip()
end = input("Destination: ").strip()

path, total_distance = a_star_search(graph, start, end)
if path:
    print(f"Path: {' -> '.join(path)}")
    print(f"Total distance: {total_distance} km")
else:
    print("NO PATH FOUND")
