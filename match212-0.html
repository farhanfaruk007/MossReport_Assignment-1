import heapq
#read file
def read_input(filename):


    graph = {}
    heuristic = {}
    with open(filename, 'r') as file:
        for line in file:
            parts = line.strip().split()
            if len(parts) < 2:
                continue
            city = parts[0]
            heuristic_val =int(parts[1])
            heuristic[city] = heuristic_val
            neighbors = []
            for i in range(2, len(parts), 2):
                neighbor = parts[i]
                distance = int(parts[i + 1])


                neighbors.append((neighbor, distance))
            graph[city] = neighbors
    return graph, heuristic

# A* search
def a_star_search(graph, heuristic, start, goal):
    open_list = []
    heapq.heappush(open_list, (heuristic[start], 0, start, [start]))  # (f_score, g_score, city, path)    


    visited = set()
    
    while open_list:
        f_score, g_score, current, path = heapq.heappop(open_list)
        if current == goal:
            return path, g_score
        
        visited.add(current)
        for neighbor, distance in graph[current]:
            if neighbor not in visited:
                new_g_score = g_score + distance
                new_f_score = new_g_score + heuristic[neighbor]
                heapq.heappush(open_list, (new_f_score, new_g_score, neighbor, path + [neighbor]))

def main(filename, start, goal):
    graph, heuristic = read_input(filename)
    path, cost = a_star_search(graph, heuristic, start, goal)
    
    if path:
        print("Most optimal value for the route Arad to Bucharest:", path)
        print("Total cost:", cost)

filename = 'input.txt'
main(filename, 'Arad', 'Bucharest')
