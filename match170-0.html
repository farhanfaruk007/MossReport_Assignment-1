import heapq

def parse_input_file(input_file):
    adjacency_list = {}
    heuristic_values = {}

    with open(input_file, 'r') as file:
        for line in file:
            elements = line.split()
            node = elements[0]
            heuristic = int(elements[1])
            heuristic_values[node] = heuristic
            neighbors = {}
            for i in range(2, len(elements), 2):
                neighbor_node = elements[i]
                weight = int(elements[i + 1])
                neighbors[neighbor_node] = weight


            adjacency_list[node] = neighbors

    return adjacency_list, heuristic_values

def a_star_algorithm(adjacency_list, heuristic_values, start_node, goal_node):
    priority_queue = []
    heapq.heappush(priority_queue, (0 + heuristic_values[start_node], 0, start_node))
    parent_map = {}
    cost_to_node = {node: float('inf') for node in adjacency_list}
    cost_to_node[start_node] = 0

    while priority_queue:
        _, current_cost, current_node = heapq.heappop(priority_queue)

        if current_node == goal_node:
            route = []
            while current_node in parent_map:
                route.append(current_node)
                current_node = parent_map[current_node]
            route.append(start_node)
            route.reverse()
            return route, current_cost

        for neighbor, edge_cost in adjacency_list[current_node].items():


            new_cost = current_cost + edge_cost
            if new_cost < cost_to_node[neighbor]:
                parent_map[neighbor] = current_node
                cost_to_node[neighbor] = new_cost
                estimated_cost = new_cost + heuristic_values[neighbor]
                heapq.heappush(priority_queue, (estimated_cost, new_cost, neighbor))



    return None, None

def main():
    input_file = '/content/24241319_S M Mahidur Rahman_CSE422_15_Lab_Assignment 01_Fall 2024.txt'
    adjacency_list, heuristic_values = parse_input_file(input_file)

    start_node = input('Enter the start node: ').strip()
    goal_node = input('Enter the destination node: ').strip()

    route, total_cost = a_star_algorithm(adjacency_list, heuristic_values, start_node, goal_node)

    if route:
        print(f'Path: {" -> ".join(route)}')
        print(f'Total distance: {total_cost} km')
    else:
        print('No path found.')

if __name__ == '__main__':
    main()
