import heapq
def make_hrstcs(inpt):
  hrstcs = {}
  for entry in inpt:
    prts = entry.strip().split()
    hrstcs[prts[0]] = int(prts[1])
  return hrstcs


def make_graph(inpt):
  ad_grph = {}
  temp_adj = {}
  for entry in inpt:
    prts = entry.strip().split()
    for i in range(2, len(prts), 2):


      temp_adj[prts[i]] = int(prts[i + 1])
    ad_grph[prts[0]] = temp_adj
    temp_adj = {}
  return ad_grph


def a_src(strt, dstntn, ad_grph, hrstcs):
  queue = []
  heapq.heappush(queue, (0, strt))
  backtrc = {strt: None}
  path_cst = {strt: 0}
  while queue:
    crnt_nd = heapq.heappop(queue)[1]
    if crnt_nd == dstntn:
      break
    for near in ad_grph[crnt_nd]:
      updtd_cst = path_cst[crnt_nd] + ad_grph[crnt_nd][near]
      if near not in path_cst or updtd_cst < path_cst[near]:
        path_cst[near] = updtd_cst


        assumed_cst = updtd_cst + hrstcs[near]
        heapq.heappush(queue, (assumed_cst, near))
        backtrc[near] = crnt_nd
  return backtrc, path_cst

def path_printer(backtrc, path_cst):
  route = []
  route.append(dstntn)
  node = dstntn
  while node != strt:
    node = backtrc[node]
    route.append(node)
  route.reverse()
  if not route:
    print("NO PATH FOUND!")
  else:
    print("Path: ", end='')
    for plc in route:
        print(plc, end=" --> " if plc != route[-1] else "\n")
    print("Total distance:", path_cst[dstntn], "km")

inpt = open("inpt.txt", "r")
hrstcs = make_hrstcs(inpt)
inpt.close()

inpt = open("inpt.txt", "r")
ad_grph = make_graph(inpt)
