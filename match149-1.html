lines = f1.readlines()
graph = {}

for line in lines:
    elements = line.split()
    node = elements[0]
    heuristic = int(elements[1])
    neighbors_data = elements[2:]

    graph[node] = {'heuristic': heuristic, 'neighbors': {}}
    for i in range(0, len(neighbors_data), 2):
        neighbor = neighbors_data[i]
        cost = int(neighbors_data[i + 1])


        graph[node]['neighbors'][neighbor] = cost

class AStarSearch:
    def __init__(self, graph):
        self.graph = graph

    def get_neighbors(self, node):
        return self.graph[node]['neighbors']

    def heuristic(self, node):
        return self.graph[node]['heuristic']

    def find_path(self, start, goal):
        open_set = set([start])
        closed_set = set()
        g_score = {start: 0}
        parent_map = {start: start}



        while open_set:
            current = None

            for node in open_set:
                if current is None or g_score[node] + self.heuristic(node) < g_score[current] + self.heuristic(current):
                    current = node

            if current is None:
                print('Path does not exist')
                return

            if current == goal:
                path = []
                total_cost = g_score[current]
                while parent_map[current] != current:
                    path.append(current)
                    current = parent_map[current]
                path.append(start)
                path.reverse()

                print(f'Path: {" -> ".join(path)}')
                print(f'Total distance: {total_cost} km')
                return



            for (neighbor, travel_cost) in self.get_neighbors(current).items():
                if neighbor not in open_set and neighbor not in closed_set:
                    open_set.add(neighbor)
                    parent_map[neighbor] = current
                    g_score[neighbor] = g_score[current] + travel_cost
                elif g_score[neighbor] > g_score[current] + travel_cost:
                    g_score[neighbor] = g_score[current] + travel_cost
                    parent_map[neighbor] = current
                    if neighbor in closed_set:
                        closed_set.remove(neighbor)
                        open_set.add(neighbor)

            open_set.remove(current)
            closed_set.add(current)

        print('Path does not exist')

# Initialize and execute the A* algorithm
a_star = AStarSearch(graph)
a_star.find_path('Arad', 'Bucharest')
