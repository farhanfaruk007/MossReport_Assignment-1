import heapq
input_file=open("/content/Input file.txt",'r')
lines=input_file.readlines()
graph={}
heuristics={}
step={}
for i in range(len(lines)):
  elem=lines[i].strip().split(' ')
  parent=elem[0]
  heuristic_value=int(elem[1])
  heuristics[parent]=heuristic_value
  child=[]
  for j in range(2,len(elem),2):
    val=(int(elem[j+1]),elem[j])
    child.append(val)
  graph[parent]=child

start=input("Start Destination: ")
end=input("End Destination: ")



heap=[]
heapq.heappush(heap,(heuristics[start],start))
node=(heuristics[start],start)
while True:
  if len(heap)==0:
    print("NO PATH FOUND")
  node=heapq.heappop(heap)
  if node[1]==end:
    break
  neighbour_list=graph[node[1]]
  for neighbour in neighbour_list:
    val,name=neighbour
    dist=node[0]-heuristics[node[1]]
    cost=dist+val+heuristics[name]
    step[(cost,name)]=node
    heapq.heappush(heap,(cost,name))
reverse_path=[node]
key=node
while key[1]!=start:
  parent=step[key]
  reverse_path.append(parent)
  key=parent
reverse_path=reverse_path[::-1]
print("\nTotal Distance:",node[0])
print("Path: ",reverse_path[0][1],end='')
for elem in reverse_path:
  if elem[1]==start:
    continue
  print(' -->',elem[1],end='')
