import heapq

city_mapping = {
    "Arad": "A", "Bucharest": "Z", "Craiova": "S", "Eforie": "T",
    "Fagaras": "O", "Dobreta": "V", "Hirsova": "N", "Iasi": "Q",


    "Neamt": "F", "Oradea": "B", "Pitesti": "P", "RimnicuVilcea": "R",
    "Timisoara": "C", "Urziceni": "D", "Vaslui": "H", "Zerind": "E",
    "Lugoj": "G", "Mehadia": "L", "Sibiu": "Sibiu", "Giurgiu": "Giurgiu"
}

heuristic_values = {}

adjacency_list = {}


def read_data(file_path):

    initial = None
    destination = None
    with open(file_path, "r") as file:

        for line in file:

            parts = line.split()
            current_city = city_mapping.get(parts[0])

            if initial is None:

                initial = current_city
            heuristic_values[current_city] = int(parts[1])

            if heuristic_values[current_city] == 0:

                destination = current_city

            adjacency_list[current_city] = []

            for i in range(2, len(parts), 2):

                neighbor_city = city_mapping.get(parts[i])
                distance = int(parts[i + 1])
                adjacency_list[current_city].append((neighbor_city, distance))

    return initial, destination

def a_star_search(initial, destination, heuristic_values, adjacency_list):

    open_list = [(0, initial)]
    path_from = {}
    cost_to_node = {initial: 0}

    while open_list:

        current_cost, current = heapq.heappop(open_list)

        if current == destination:

            break

        for neighbor, travel_cost in adjacency_list.get(current, []):

            new_cost = cost_to_node[current] + travel_cost

            if neighbor not in cost_to_node or new_cost < cost_to_node[neighbor]:

                cost_to_node[neighbor] = new_cost
                priority = new_cost + heuristic_values.get(neighbor, float('inf'))
                heapq.heappush(open_list, (priority, neighbor))
                path_from[neighbor] = current

    return path_from, cost_to_node

def generate_output(path_from, cost_to_node, initial, destination):

    route = []
    current = destination

    while current != initial:

        if current not in path_from:
            print("NO PATH FOUND")
            return

        route.append(current)
        current = path_from[current]
    route.append(initial)
    route.reverse()


    reverse_city_mapping = {v: k for k, v in city_mapping.items()}
    full_route = " -> ".join(reverse_city_mapping.get(node, f"Unknown({node})") for node in route)

    print(f"Route: {full_route}")
    print(f"Total distance: {cost_to_node.get(destination, 'N/A')} km")

file_path = "input1.txt"
initial, destination = read_data(file_path)
if initial is None or destination is None:
    print("NO PATH FOUND")

else:
    path_from, cost_to_node = a_star_search(initial, destination, heuristic_values, adjacency_list)
    generate_output(path_from, cost_to_node, initial, destination)
