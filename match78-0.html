import heapq



def aStarAlgo(sPoint, ePoint, graph, heuristic):
    fringe = []
    heapq.heapify(fringe)
    heapq.heappush(fringe, (0, sPoint))
    path = {}
    path[sPoint] = None
    distance = {}
    distance[sPoint] = 0
    while fringe:
        cNode = heapq.heappop(fringe)[1]
        if cNode == ePoint:
            break
        for i in graph[cNode]:
            temp = distance[cNode] + int(graph[cNode][i])
            if i not in distance or temp < distance[i]:
                distance[i] = temp
                totalDistance = int(heuristic[i]) + temp
                heapq.heappush(fringe, (totalDistance, i))
                path[i] = cNode
    return path, distance

def printPath(sPoint, ePoint):


    if (sPoint and ePoint in heuristic):
        path, distance = aStarAlgo(sPoint, ePoint, graph, heuristic)    
        actualPath = [ePoint]
        child = ePoint
        while child != sPoint:
            parent = path[child]
            actualPath.append(parent)
            child = parent
        if len(actualPath) == 0:
            print("NO PATH FOUND!")
        else:
            actualPath.reverse()
            printAns = " -> ".join(actualPath)
            print(f"Path: {printAns} \nTotal distance: {distance[ePoint]} km")
    else:
        print("NO PATH FOUND!")

inputData = open("input.txt", "r")
graph = {}
heuristic = {}
currNode = inputData.readline().strip()

while currNode != "":
    cNodeList = currNode.split(" ")
    for i in range(0, len(cNodeList), +2):
        if i == 0:
            heuristic[cNodeList[0]] = int(cNodeList[1])
            graph[cNodeList[0]] = {}
        else:
            graph[cNodeList[0]][cNodeList[i]] = int(cNodeList[i+1])
    currNode = inputData.readline().strip()

sPoint = input("Starting Point: ")
ePoint = input("Destination Point: ")

printPath(sPoint, ePoint)
