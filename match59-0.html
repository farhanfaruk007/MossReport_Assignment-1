import heapq

def r_inpFile(inpFile):
    g = {}
    h = {}

    with open(inpFile, 'r') as inpFile:
        for l in inpFile:
            temp = l.split()


            p_city = temp[0]
            c_heuristic = int(temp[1])
            childs = temp[2:]

            g[p_city] = {}
            for i in range(0, len(childs), 2):
                c = childs[i]
                distance = int(childs[i+1])
                g[p_city][c] = distance

            h[p_city] = c_heuristic


    return g, h

def a_star(g, h, s_state, g_state):
    open_list = []
    heapq.heappush(open_list, (0, s_state))
    saga = {}
    g_cost = {s_state: 0}
    f_cost = {s_state: h[s_state]}

    while open_list:
        curr = heapq.heappop(open_list)[1]

        if curr == g_state:
            path = []
            while curr in saga:
                path.append(curr)
                curr = saga[curr]
            path.append(s_state)
            path.reverse()
            return path, g_cost[g_state]

        for c in g[curr]:
            approx_g_cost = g_cost[curr] + g[curr][c]

            if c not in g_cost or approx_g_cost < g_cost[c]:
                saga[c] = curr
                g_cost[c] = approx_g_cost
                f_cost[c] = approx_g_cost + h[c]
                heapq.heappush(open_list, (f_cost[c], c))

    return None, None
