from queue import PriorityQueue


# Open the file 'input.txt' in read mode
with open('input.txt', 'r') as n:
    dict = {}

    # 20 ta input deya
    for _ in range(20):


        arr = n.readline().split()  # eita te prottek ta line ache
        #print(hold)
        # The first part is the node name and the second is its heuristic
        dict[arr[0]] = [int(arr[1]), []]
        #print(dict)

        for j in range(2, len(arr), 2):

            dict[arr[0]][1].append((arr[j], int(arr[j + 1])))#connected nodes
    #print(dict)







start='Arad'
end='Bucharest'




que = PriorityQueue()
temp= dict[start]
#print(temp)
que.put([temp[0],[start],0])


#final result
def pathsearch(array,cost):


  final_path=""
  for i in range(len(array)):
    #print(array)
    if i!=len(array)-1:
      final_path+=array[i]+"--> "
    else:
      final_path+=array[i]
  print('Path:',final_path)
  print("Total Distance:",cost)
  path_paisi=True





visited=[]
path_paisi=False
while not que.empty():
  deque= que.get()

  node=deque[1][len(deque[1])-1]
  cost=deque[2]
  visited.append(node)

  if node==end:
    path_paisi=True

    pathsearch(deque[1],cost)
    break



  for tuplee in dict[node][1]:
    #print(tup)

    if tuplee[0] not in visited:
      pathcost=tuplee[1]+cost
      #print(pathcost)
      huristic=dict[tuplee[0]][0]
      #print(huristic)
      totalcost=huristic+ pathcost


      copy=deque[1].copy()
      new=[totalcost,copy,pathcost]




      new[1].append(tuplee[0])
      que.put(new)
if path_paisi==False:
  print("NO PATH FOUND")
