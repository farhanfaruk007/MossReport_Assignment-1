import math
from queue import PriorityQueue

def read_file(file_path):
    heuristic = {}
    graph = {}

    with open('input.txt', 'r') as f:
        lines = f.read().strip().split("\n")
    i = 0
    while i < len(lines):

        line = lines[i].strip()
        parts = line.split()

        if len(parts) < 3:
            i += 1
            continue

        node = parts[0]
        heuristic_value = int(parts[1])
        heuristic[node] = heuristic_value

        edges = []
        j = 2
        while j < len(parts):
            neighbor = parts[j]
            distance = int(parts[j + 1])
            edges.append((neighbor, distance))
            j += 2
        graph[node] = edges
        i += 1


    print("Graph:", graph)
    print("Heuristics:", heuristic)

    return heuristic, graph

def astar_search(graph, start, destination, heuristic):
    queue = PriorityQueue()
    parent = {}
    dist = {}
    total_cost = {}



    for node in graph:
        dist[node] = math.inf
        total_cost[node] = math.inf
        parent[node] = None

    dist[start] = 0
    total_cost[start] = heuristic[start]
    queue.put((total_cost[start], start))

    while not queue.empty():
        curr_cost, curr_node = queue.get()

        if curr_node == destination:
            path = []
            while curr_node:


                path.append(curr_node)
                curr_node = parent[curr_node]
            path.reverse()
            return path, dist[destination]

        for neighbor, weight in graph[curr_node]:
            new_dist = dist[curr_node] + weight
            total_neighbor_cost = new_dist + heuristic[neighbor]
            if new_dist < dist[neighbor]:
                dist[neighbor] = new_dist
                parent[neighbor] = curr_node
                queue.put((total_neighbor_cost, neighbor))

    return "NO PATH FOUND", 0

file_path = "\input.txt"

heuristic, graph = read_file(file_path)

start_node = input("Start Node: ").strip().capitalize()
destination_node = input("Destination Node: ").strip().capitalize()

pathway, cost = astar_search(graph, start_node, destination_node, heuristic)

if cost != 0:
    print("Path:", ' -> '.join(pathway))
    print(f'Total distance: {cost} km')
else:
    print(pathway)
