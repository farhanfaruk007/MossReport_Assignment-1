import heapq

question = open("aStarGraph", "r")

def howCloseToGoal(question):

#heuristic value of the current node

    h1 = {}

#heuristic value of the neighbouring nodes

    h2 = {}

#used for the path cost calculation by heuristic values(temporary storage for heusistic value of the neighbours of the node)

    heuCalMed = {}

#slicing for every lines in input file

    for x in question:

#names of cities

        y = x.strip().split()

#numbers of cities

        z = len(y)

#heuristic value of the node itself

        h1[y[0]] = int(y[1])

#heuristic value of connected cities,that is,the neighbouring nodes

        for i in range(2, z, 2):

          heuCalMed[y[i]] = int(y[i+1])

        h2[y[0]] = heuCalMed

        heuCalMed = {}

    return (h1, h2)

#obtaining data for heuristic values



h1, h2= howCloseToGoal(question)

# A-star algorithm begins ( heuristic magnitude + actual path magnitude )

def pathCostCalculation(initialNode, goalNode, h2, h1):

#An array to track the visited cities(we are using heap data structure)

    a = []

#Prioirity queue is being used .Initially pushes with value of priority as 0

    heapq.heappush(a, (0, initialNode))

    distanceBetweentheNodes = {}

#tracing the cost of shortest path from initial node to current node

    magnitude = {}

#Tracks the parent node for reconstructing the path

    distanceBetweentheNodes[initialNode] = None

    magnitude[initialNode] = 0

    while a:

        presentNode = heapq.heappop(a)[1]

        if presentNode == goalNode:

            break

#it continues with the lowest priority in the heap till the end of the search/goal node is reached

        for children in h2[presentNode]:

            new_magnitude = magnitude[presentNode] + h2[presentNode][children]

            if children not in magnitude or new_magnitude < magnitude[children]:

                magnitude[children] = new_magnitude

                hV = new_magnitude + h1[children]



                heapq.heappush(a, (hV, children))

                distanceBetweentheNodes[children] = presentNode

    return distanceBetweentheNodes, magnitude

#Calculating the final results

def result(path, magnitude, initialNode, goalNode):

    presentNode = goalNode

    distanceBetweentheNodes = []

    while presentNode is not initialNode:

      distanceBetweentheNodes.append(presentNode)

      newNode = path[presentNode]

      presentNode = newNode

    distanceBetweentheNodes.append(initialNode)

    if len(distanceBetweentheNodes) != 0:

      distanceBetweentheNodes.reverse()

      print(*distanceBetweentheNodes, sep = " --> ")

    else:

      print(" NO PATH FOUND ")

    print(f'Total distance: {magnitude[goalNode]} km')


initialNode = input("Start node: ")

goalNode = input("Destination: ")

if initialNode and goalNode not in h1:

    print("Incorrect Search")

else:
    path, magnitude = pathCostCalculation(initialNode, goalNode, h2, h1)

    result(path, magnitude, initialNode, goalNode)
