import heapq
def A_star(graph,strt,destination):
    priority_queue=[(graph[strt]['heuristic'], strt)]
    visited_nodes=set()
    predecessors={}
    g_val= {node: float('inf') for node in graph}
    g_val[strt]=0


    while priority_queue:
        _, cur_node = heapq.heappop(priority_queue)


        if cur_node==destination:
            path=[]
            while cur_node in predecessors:
                path.insert(0, cur_node)
                cur_node = predecessors[cur_node]
            path.insert(0, strt)
            return path
        visited_nodes.add(cur_node)

        for neighbor, distance in graph[cur_node]['neighbors'].items():
            new_g_val = g_val[cur_node]+ distance


            if new_g_val<g_val[neighbor]:
                predecessors[neighbor] = cur_node
                g_val[neighbor] = new_g_val
                total_cost_estimated = new_g_val + graph[neighbor]['heuristic']
                heapq.heappush(priority_queue, (total_cost_estimated, neighbor))

    return None


graph = {}
with open(("input.txt"),'r') as file:
  for i in file:
    lines = i.split()
    node = lines[0]
    h_val = int(lines[1])
    neighbors = {}
    for i in range(2, len(lines), 2):


        neighbor =lines[i]
        distance = int(lines[i + 1])
        neighbors[neighbor] = distance
    graph[node]={'heuristic':h_val, 'neighbors':neighbors}

f2= open("output.txt", 'w')



start_node=input("Enter start node: ")
destination_node=input("Enter destination node: ")

if start_node not in graph:
    print("No start node is given.")
if destination_node not in graph:
    print("No destination node is given.")

test = A_star(graph, start_node, destination_node)
if test:
    total_distance = sum(graph[node]['neighbors'][test[i + 1]] for i, node in enumerate(test[:-1]))
    print("Path:"," -> ".join(test), file=f2)
    print("Total distance:", total_distance, "km", file=f2)
else:
    print("NO PATH FOUND", file=f2)


f2.close()
