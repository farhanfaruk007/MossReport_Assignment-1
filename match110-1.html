import heapq
import math
inpfile = open("/content/22101407_Lab01_CSE422.txt","r")

heuristicevalue = {}
citygrph = {}

mapping = {'Arad':'A','Bucharest':'Z','Craiova':'S','Eforie':'T','Fagaras':'O','Dobreta':'V','Hirsova':'N','lasi':'Q',
                'Neamt':'F','Oradea':'B','Pitesti':'P','RimnicuVilcea':'R','Timisoara':'C','Urziceni':'D','Vaslui':'H','Zerind':'E','Lugoj':'G','Mehadia':'L','Sibiu':'I','Giurgiu':'M'}

reversedmapping = {value:key for key,value in mapping.items()}

#graphcreation


for line in inpfile:
        parts = line.split()
        city = parts[0]
        abbreviation = mapping[city]

        heuristic = int(parts[1])
        heuristicevalue[abbreviation] = heuristic

        if abbreviation not in citygrph:
            citygrph[abbreviation] = []

        neighbors = parts[2:]
        for i in range(0, len(neighbors), 2):
            neighbor_city = neighbors[i]
            distance = int(neighbors[i + 1])
            neighbor_abbreviation = mapping[neighbor_city]


            citygrph[abbreviation].append((neighbor_abbreviation, distance))

#A*create
def Astar(graph,heu,start,d):
  pqueue = []
  heapq.heappush(pqueue,(heu[start],start))
  gscores ={start:0}
  fscores = {start: heu[start]}
  visitedtrack = set()
  parent_node = {start: None}



  while pqueue:

    curr_val,current = heapq.heappop(pqueue)

    if current == d:
      reqpath = []
      while current:
        reqpath.append(current)
        current = parent_node[current]
      reqpath.reverse()

      return reqpath,gscores[d]

    if current in visitedtrack:
      continue
    visitedtrack.add(current)


    for i,j in graph[current]:


      if i in visitedtrack:
        continue
      gscorecalculation = gscores[current] + j

      if gscorecalculation < gscores.get(i,math.inf):
        parent_node[i] = current
        gscores[i] = gscorecalculation
        fscores[i] = gscorecalculation + heu[i]
        heapq.heappush(pqueue,(fscores[i],i))

  return None,None


startnode = input("Start node: ").strip()
endnode = input("Destination: ").strip()

start = mapping[startnode]
destination = mapping[endnode]

trackpath, total_dist = Astar(citygrph,heuristicevalue,start,destination)
if trackpath:
  actual_path = [reversedmapping[i] for i in trackpath]
  print("Path:",end=" ")
  print("->".join(actual_path))
  print("Total Distance:", total_dist, "km")
else:
  print("NO PATH FOUND")
