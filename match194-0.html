graph = {}
h_value = {}
input1 = open(file='/content/drive/MyDrive/cse422/input.txt', mode='r')
for l in input1:
  data = l.split()
  c = data[0]
  h = int(data[1])
  h_value[c] = h
  graph[c] = []
  for i in range(2, len(data), 2):
    n = data[i]
    dist = int(data[i+1])
    graph[c].append((n, dist))

def get_neighbor(v):
  if v in graph:
    return graph[v]
  else:
    return None

def astaralgo(s,e):
  new = set([s])
  explored = set()
  d = {}
  parent = {}
  d[s] = 0
  parent[s] = s



  while len(new) > 0:
    n = None
    for i in new:
      if n == None or d[i] + h_value[i] < d[n] + h_value[n]:
        n = i
    if n == e or graph.get(n) == None:
      pass
    else:
      for (m,w) in get_neighbor(n):
        if m not in new and m not in explored:
          new.add(m)
          parent[m]=n
          d[m] = d[n] + w
        else:
          if d.get(m, float('inf')) > d[n] + w:


            d[m] = d[n] + w
            parent[m] = n
            if m in explored:
              explored.remove(m)
              new.add(m)
    if n == None:
      print("Path doesn't exist")
      return None
    if n == e:
      path=[]
      td = d[n]
      while parent[n] != n:
        path.append(n)
        n = parent[n]
      path.append(s)
      path.reverse()
      #print(f'Path found: {path}')
      return path, td
    new.remove(n)
    explored.add(n)
  print('Path does not exist')
  return None

path,td=astaralgo(next(iter(graph)), next(reversed(graph)))

if path != None:
    print(f"Path: {' -> '.join(path)}")
    print(f"Total distance: {td} km")
else:
    print("No path found")
