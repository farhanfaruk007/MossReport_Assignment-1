import heapq

def read_graph(file_path):
    graph = {}  #adj list of the graph
    h = {}  #storing all the heuristic val
    with open(file_path, 'r') as file:
        for line in file:


            parts = line.strip().split()
            node = parts[0]
            h[node] = int(parts[1])  #extract heuristic val from the node
            graph[node] = [(parts[i], int(parts[i + 1])) for i in range(2, len(parts), 2)] #tuples of neighbor & cost
    return graph, h

def a_star(graph, h, start, goal):
    pq = [(h[start], 0, start, [])]
    visited = set()  #track visited nodes

    while pq:
        temp, g, current, path = heapq.heappop(pq)  #node with smallest total estimated cost value

        if current in visited:
            continue
        path = path + [current]

        if current == goal:
            return path, g  #return the path and cost

        visited.add(current)  #mark the current node as visited



        for neighbor, cost in graph.get(current, []):  #explore neighbors
            if neighbor not in visited:
                f = g + cost + h[neighbor]  #total estimated cost
                heapq.heappush(pq, (f, g + cost, neighbor, path))

    return None, None

def search_path():
    graph, h = read_graph("input_file.txt")
    start = input("Enter the starting node: ")
    goal = input("Enter the goal node: ")
    path, cost = a_star(graph, h, start, goal)

    if path:
        print(f"Path: {' -> '.join(path)}\nTotal Cost: {cost} km")
    else:
        print("No Path Found")

search_path()
