open_info = open("Input file.txt","r")

# starting_from = "Arad"
# ending_at = "Bucharest"
starting_from = input("Please enter the starting point: ")
ending_at = input("Please enter the ending point: ")

h_dict_table = {}
adj_dict_table = {}
path_status = True
li_of_parent = []
li_of_parent_idx = []

given_info = open_info.readlines()



for p in range(len(given_info)) :
    line = given_info[p]
    elem = line.split()
    h_dict_table[elem[0]] = int(elem[1])
    adj_dict_table[elem[0]] = []
    for j in range(2, len(elem), 2):
        adj_dict_table[elem[0]].append([elem[j],int(elem[j+1])])

# print("========================================================")
# print(h_dict_table)
# print("Neighbour: ", adj_dict_table)

## A* search  ##



new_list = []
s = starting_from
h_of_s = h_dict_table[starting_from]
new_list.append([h_of_s, s])
pop_node = None

while pop_node != ending_at :

    if len(new_list) == 0 :
        path_status = False
        break
    pop = new_list.pop(0)

    pop_node = pop[1]
    pop_value = pop[0]
    neighbor = adj_dict_table[pop_node]
    # print("PPP", neighbor)

    for city in neighbor:
        nam_of_city = city[0]
        city_dis_self = city[1]
        count_dis = pop_value - h_dict_table[pop_node]
        dis_of_city = count_dis + city_dis_self + h_dict_table[nam_of_city]

        if nam_of_city not in li_of_parent_idx :
            new_list.append([dis_of_city, nam_of_city])
            li_of_parent_idx.append(nam_of_city)
            li_of_parent.append(pop_node)
        else :
            for k in range(len(new_list)):
                if nam_of_city == new_list[k][1] :
                    pp =  new_list[k][0]
                    idx = k
                    if pp > dis_of_city :
                        new_list.pop(idx)
                        new_list.append([dis_of_city, nam_of_city])
                        s = li_of_parent_idx.index(nam_of_city)
                        li_of_parent_idx.pop(s)
                        li_of_parent.pop(s)
                        li_of_parent_idx.append(nam_of_city)
                        li_of_parent.append(pop_node)

    new_list.sort()
    # print("New_List: ", new_list)

# print("Total New_List: ", new_list)

# print("Li_of_Parent: ", li_of_parent)
# print("Li_of_Parent_idx: ", li_of_parent_idx)

if path_status == False :
    print("NO PATH FOUND!!!")
else :
    path = []
    path.append(ending_at)

    father = None

    while father != starting_from :

        idx = li_of_parent_idx.index(ending_at)
        father = li_of_parent[idx]
        path.append(father)
        ending_at = father

    path = path[::-1]
    print("Path:", " -> ".join(path))
    print(f"Total distance: {pop_value} km")
