import heapq

def Aestar_line(start, goal, adj_graph, est):
    line_mr = []
    heapq.heappush(line_mr, (0, start))
    former = {}
    wgt = {}
    former[start] = None
    wgt[start] = 0



    while line_mr :
        recent = heapq.heappop(line_mr)[1]

        if recent == goal:
            break

        for positive_r in adj_graph[recent]:
            new_wgt= wgt[recent] + adj_graph[recent][positive_r]
            if positive_r not in wgt or new_wgt< wgt[positive_r]:
                wgt[positive_r] = new_wgt


                priority = new_wgt + est[positive_r]
                heapq.heappush(line_mr, (priority, positive_r))
                former[positive_r] = recent
    return former , wgt
