class City:
    def __init__(self, name, heuristic):
        self.name = name
        self.heuristic = heuristic
        self.neighbors = {}

    def add_neighbor(self, neighbor_name, distance):
        self.neighbors[neighbor_name] = distance

def read_map_from_file(filename):
    cities = {}
    file = open(filename, 'r')
    lines = file.readlines()
    line_index = 0
    while line_index < len(lines):
        parts = lines[line_index].strip().split()
        city_name = parts[0]
        heuristic = int(parts[1])
        city = City(city_name, heuristic)
        i = 2
        while i < len(parts):
            neighbor = parts[i]
            distance = int(parts[i + 1])
            city.add_neighbor(neighbor, distance)
            i += 2
        cities[city_name] = city
        line_index += 1
    file.close()


    return cities

class Algorithm:
    def __init__(self, adjacency_list):
        self.adjacency_list = adjacency_list

    def get_neighbors(self, node):
        return self.adjacency_list[node].neighbors

    def heuristic(self, node):
        return self.adjacency_list[node].heuristic

    def a_star(self, start, end):
        open_list = set([start])
        closed_list = set([])
        g_score = {start: 0}
        parents = {start: start}



        while len(open_list) > 0:
            current_node = None


            for node in open_list:
                if current_node is None or g_score[node] + self.heuristic(node) < g_score[current_node] + self.heuristic(current_node):
                    current_node = node

            if current_node is None:
                print('Path does not exist')
                return


            if current_node == end:
                path = []
                total_cost = g_score[current_node]
                while parents[current_node] != current_node:
                    path.append(current_node)
                    current_node = parents[current_node]
                path.append(start)
                path.reverse()


                path_string = " -> ".join(path)
                print(f'Path: {path_string}')
                print(f'Total distance: {total_cost} km')
                return


            neighbors = list(self.get_neighbors(current_node).items())
            i = 0
            while i < len(neighbors):


                neighbor, weight = neighbors[i]
                if neighbor not in open_list and neighbor not in closed_list:
                    open_list.add(neighbor)
                    parents[neighbor] = current_node
                    g_score[neighbor] = g_score[current_node] + weight
                else:
                    if g_score[neighbor] > g_score[current_node] + weight:
                        g_score[neighbor] = g_score[current_node] + weight
                        parents[neighbor] = current_node
                        if neighbor in closed_list:
                            closed_list.remove(neighbor)
                            open_list.add(neighbor)
                i += 1


            open_list.remove(current_node)
            closed_list.add(current_node)

        print('Path does not exist')
