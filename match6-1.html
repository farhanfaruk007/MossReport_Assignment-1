inputf =open("input.txt", "r")



inp_lines = inputf.readlines()

H_values={}
G_dict={}


def checkeDict(i):
    if i in G_dict:
        return G_dict[i]
    else:
        return None
    
    
for i in range(len(inp_lines)):
    inp_lines[i]=inp_lines[i][:len(inp_lines[i])-1]

def AstarSearch(st_node,end_node):
    v_nodes =set()
    n_nodes =set()
    v_nodes.add(st_node)

    dict_map ={}
    total_cost ={}
    total_cost[st_node]=0
    dict_map[st_node]=st_node


    while len(v_nodes)>0:
        x= None

        for i in v_nodes:
            if x is None or total_cost[i]+H_values[i]<total_cost[x]+H_values[x]:
                x=i

        if x==end_node or G_dict[x]==None:
            pass

        else:
            for (next_node, c) in checkeDict(x):
                if next_node not in v_nodes and next_node not in n_nodes:
                    v_nodes.add(next_node)
                    dict_map[next_node]=x
                    total_cost[next_node]=total_cost[x]+c
                    p_distance=total_cost[next_node]

                else:
                    if total_cost[next_node]>total_cost[x]+c:
                        total_cost[next_node]=total_cost[x]+c
                        dict_map[next_node]=x

                        if next_node in n_nodes:
                            n_nodes.remove(next_node)
                            v_nodes.add(next_node)
                        p_distance=total_cost[next_node]

        if x==None:
            print('No path found')
            return None
    
        if x==end_node:
            temp_list=[]

            while dict_map[x]!=x:
                temp_list.append(x)
                x=dict_map[x]

            temp_list.append(st_node)
            temp_list.reverse()

            # print(temp_list)
            # return p_distance
            print(*temp_list, sep=" -> ")
            print("Total_distance:", p_distance, "km")
            return p_distance
        v_nodes.remove(x)
        n_nodes.add(x)

    print('No path found') 
    return None
