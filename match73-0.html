import heapq

class Graph:
    def __init__(self):
        self.graph={}

    def node(self,node,h,n):
        self.graph[node]= {'h':h,'n':n}



    def Astar_search(self,start,goal):
        open_list= []
        h_costs = {node: float('inf') for node in self.graph}
        h_costs[start] = 0
        pre_node = {}



        heapq.heappush(open_list, (h_costs[start], start))

        while open_list:
            current_cost, current_node = heapq.heappop(open_list)

            if current_node == goal:
                path = []
                while current_node in pre_node:
                    path.append(current_node)
                    current_node = pre_node[current_node]
                path.append(start)
                path.reverse()
                return path, h_costs[goal]


            for neighbor, cost in self.graph[current_node]['n']:
                total_cost = current_cost + cost
                if total_cost < h_costs[neighbor]:
                    h_costs[neighbor] = total_cost
                    pre_node[neighbor] = current_node


                    heapq.heappush(open_list, (total_cost, neighbor))

        return 'No Path Found', None

def load_graph(file_name):
    graph = Graph()
    with open(file_name, 'r') as file:
        for line in file:
            parts= line.strip().split()
            if len(parts) < 2:
                continue
            node= parts[0]
            h= int(parts[1])


            n= [(parts[i], int(parts[i + 1])) for i in range(2, len(parts), 2)]
            graph.node(node,h,n)
    return graph

def main():
    file_name='input.txt'
    graph=load_graph(file_name)

    start=input('Enter Start Node: ')
    goal=input('Enter Goal Node: ')

    path, distance= graph.Astar_search(start,goal)

    if path =='No Path Found':
        print('No Path Found')
    else:
        print('Path:', path)
        print('Distance:', distance)

if __name__ == '__main__':
    main()
