import heapq


def a_star_search(start, target, grph_info, heuristic_valss):
    prior_queue = []
    heapq.heappush(prior_queue, (0, start))
    prev_node = {}
    updt_cst = {}
    prev_node[start] = None
    updt_cst[start] = 0

    while prior_queue:
        on_going = heapq.heappop(prior_queue)[1]

        if on_going == target:
            break

        for neighbor in grph_info[on_going]:
            new_cst = updt_cst[on_going] + grph_info[on_going][neighbor]
            if neighbor not in updt_cst or new_cst < updt_cst[neighbor]:
                updt_cst[neighbor] = new_cst


                total_cost = new_cst + heuristic_valss[neighbor]
                heapq.heappush(prior_queue, (total_cost, neighbor))
                prev_node[neighbor] = on_going

    return prev_node, updt_cst
