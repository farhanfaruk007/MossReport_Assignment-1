import heapq

def readFile(filename):
  graph = {}
  heuristic_values = {}

  with open(filename, 'r') as file:
    for line in file:
      parts = line.split()


      city = parts[0]
      heuristic_value = int(parts[1])
      heuristic_values[city] = heuristic_value
      neighbors = parts[2:]
      graph[city] = {}
      for i in range(0,len(neighbors),2):
        neighbor = neighbors[i]
        distance = int(neighbors[i+1])
        graph[city][neighbor] = distance

  return graph, heuristic_values

def aStar(graph, heuristic_values, start, goal):
  priority_queue = [(heuristic_values[start], 0, start,[start])]
  heapq.heapify(priority_queue)
  visited = set()

  while priority_queue:
    total_cost, current_cost, current_city, path = heapq.heappop(priority_queue)

    if current_city == goal:
      print("Path:","->".join(path))
      print("Total Distance:", current_cost,"Km")
      return

    visited.add(current_city)

    for neighbor, distance in graph[current_city].items():
      if neighbor in visited:
        continue

      new_cost = current_cost + distance
      new_total_cost = new_cost + heuristic_values[neighbor]
      new_path = path + [neighbor]

      heapq.heappush(priority_queue, (new_total_cost, new_cost, neighbor, new_path))
  print("NO PATH FOUND")

filename = 'Input file.txt'
graph, heuristic_values = readFile(filename)
start = 'Arad'
goal = 'Bucharest'
aStar(graph, heuristic_values, start, goal)
