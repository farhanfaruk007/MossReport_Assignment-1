import heapq

def parse_input(file_path):
    graph = {}
    heuristics = {}
    with open(file_path, 'r') as file:
        for line in file:
            parts = line.split()
            city = parts[0]
            heuristics[city] = int(parts[1])
            neighbors = [(parts[i], int(parts[i+1])) for i in range(2, len(parts), 2)]
            graph[city] = neighbors


    return graph, heuristics


def a_star_search(graph, heuristics, start, end):
    open_set = []
    heapq.heappush(open_set, (0 + heuristics[start], 0, start, [start]))  # (f, g, current_city, path)
    visited = set()

    while open_set:
        f, g, current, path = heapq.heappop(open_set)

        if current == end:
            return path, g  # path and total distance

        visited.add(current)

        for neighbor, distance in graph.get(current, []):
            if neighbor in visited:
                continue
            new_g = g + distance


            new_f = new_g + heuristics[neighbor]
            heapq.heappush(open_set, (new_f, new_g, neighbor, path + [neighbor]))

    return None, None  # No path found


def main():

    graph, heuristics = parse_input("/content/Input file.txt")


    start = input("Start city: ").strip() #Arad #CASE SENSITIVE
    destination = input("Destination city: ").strip() #Bucharest #CASE SENSITIVE
    # print(f"Graph: {graph}")
    # print(f"heuristics: {heuristics}")
    path, total_distance = a_star_search(graph, heuristics, start, destination)

    if path:
        print("Path:", " -> ".join(path))
        print("Total distance:", total_distance, "km")
    else:
        print("NO PATH FOUND")


main()
