f1 = open("/content/Input_file.txt", "r")
f2 = open("/content/output_1a.txt", "w")
lines = f1.readlines()



nodes = []
heuristics = []
connections = []

for line in lines:
    words = line.strip().split()
    nodes.append(words[0])
    heuristics.append(int(words[1]))
    connection_info = words[2:]


    for idx, item in enumerate(connection_info):
        if item.isdigit():
            connection_info[idx] = int(item)
    connections.append(connection_info)


start_node, goal_node = input("Start node: "), input("Destination: ")



open_set = {start_node}
came_from = {}
g_score = {node: float('inf') for node in nodes}
g_score[start_node] = 0


# f_score = g(n) + h(n)
f_score = {node: float('inf') for node in nodes}
f_score[start_node] = heuristics[nodes.index(start_node)]


while open_set:

    current_node = min(open_set, key=lambda x: f_score[x])



    if current_node == goal_node:

        path = []
        total_cost = g_score[goal_node]
        while current_node in came_from:
            path.insert(0, current_node)
            current_node = came_from[current_node]
        path.insert(0, start_node)


        path_str = " -> ".join(path)
        print(path_str)
        print("Total distance:", total_cost, "km")

        f2.write(path_str + "\n")
        f2.write("Total distance: " + str(total_cost) + " km")
        break

    open_set.remove(current_node)
    node_index = nodes.index(current_node)


    neighbors = connections[node_index]

    for i in range(0, len(neighbors), 2):
        neighbor = neighbors[i]
        travel_cost = neighbors[i+1]


        tentative_g_score = g_score[current_node] + travel_cost
        if tentative_g_score < g_score[neighbor]:

            came_from[neighbor] = current_node
            g_score[neighbor] = tentative_g_score
            f_score[neighbor] = g_score[neighbor] + heuristics[nodes.index(neighbor)]
            if neighbor not in open_set:
                open_set.add(neighbor)


else:

    print("No path found")
    f2.write("No path found")


# Closing files
f1.close()
f2.close()
