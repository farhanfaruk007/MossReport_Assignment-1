from queue import PriorityQueue
heuristic_values = {}
path = {}
file1 = open('input.txt','r')
file2 = open('output.txt','w')
x = file1.readlines()
for i in x:
  i = i.split()


  path[i[0]] = []
  for j in range(2,len(i),2):
      path[i[0]].append([i[j],int(i[j+1])])
  heuristic_values[i[0]] = int(i[1])




def A_star(start,goal,heuristic_values,path):
  open = PriorityQueue()
  open.put((0,start))
  visited = []
  parent = {start:start}
  g_score = {start:0}
  f_score = {start:0+heuristic_values[start]}

  while open.empty() == False:
    pop = open.get()
    lowest_f_score = pop[0]
    node = pop[1]

    if node == goal:
      return (node,lowest_f_score,parent)

    visited.append(node)


    for child,weight in path[node]:
      if child not in visited:
        parent[child] = node

      if child in visited:
        continue



      g_score[child] = g_score[node] + weight
      f_score[child] = g_score[child] + heuristic_values[child]
      open.put((f_score[child],child))

  if open.qsize() == 0:
    return "NO PATH FOUND"


start_node = input("Start node: ")
destination = input("Destination: ")
ans = A_star(start_node,destination,heuristic_values,path)
if ans == "NO PATH FOUND":
  print(ans)


  print(ans, file = file2)
else:
  parent = ans[2]
  optimal_path = []
  while destination != start_node:
    optimal_path.append(destination)
    destination = parent[destination]
  optimal_path.append(start_node)
  optimal_path.reverse()
  print('Path: ',end="", file = file2)
  for city in optimal_path:
    if optimal_path.index(city) != (len(optimal_path)-1):
       print(city,'->',end=" ",file = file2)
    else:
      print(city, file = file2)

  print('Total distance:',ans[1],'km', file = file2)
