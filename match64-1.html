import heapq

x = open("Input file.txt", 'r')
li = x.readlines()

graph={}
est_cost={}
path_dis={}
for i in range(len(li)):
  nodes=li[i].strip().split(' ')
  par=nodes[0]
  heu_cost=int(nodes[1])
  est_cost[par]=heu_cost
  neib=[]
  for j in range(2,len(nodes),2):
    n=(nodes[j], int(nodes[j+1]))
    neib.append(n)
  graph[par]=neib

# for i in graph.items():
#   print(i)
# for i in est_cost.items():
#   print(i)

def A_search(s, g):


    q=[]
    heapq.heappush(q,(est_cost[s],s))
    node=(est_cost[s],s)
    while True:
      if len(q)==0:
        print("NO PATH FOUND")
      
      node=heapq.heappop(q)
      if node[1]==g:
        break
      
      neibs=graph[node[1]]
      for i in neibs:
        n, val=i
        dist=node[0]-est_cost[node[1]]
        total=dist+val+est_cost[n]
        path_dis[(total,n)]=node
        heapq.heappush(q,(total,n))
    path=[node]
    final=node
    # print(final)
    while final[1]!=s:
        x=path_dis[final]
        path.append(x)
        final=x
    path=path[::-1]

    # for i in path_dis.items():
    #     print(i)

    print("\nTotal Distance:",node[0])
    print("Path: ",end='')
    # print(path)
    for i in range(len(path)):
        if i!=len(path)-1:
            print(path[i][1],'-> ',end='')
        else:
            print(path[i][1],end='')

Start = input()
Goal = input()
A_search(Start, Goal)

x.close()
