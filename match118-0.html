from queue import PriorityQueue


def A_Star_Search(start, goal, heuristic,dist_path):
    que = PriorityQueue()
    que.put((0, start))
    visited = []
    parent={start:start}
    cost_initial={start:0}
    cost_after=0+heuristic[start]
    total={start:cost_after}

    while que.empty() == False:
        score=que.get()
        cost=score[0]
        node=score[1]

        if node in visited:
            continue

        visited.append(node)
        for key, value in dist_path[node]:
            new_cost= cost_initial[node]+value
            if key in visited:
                continue
            if key not in visited:
                parent[key]=node


            cost_initial[key]=cost_initial[node]+value
            total[key]=cost_initial[key]+heuristic[key]
            que.put((total[key],key))

            # if new_cost < cost_initial[value]:

            #     que.put((total[key],key))
            #     parent[key]=node

        if node == goal:
            return (node,cost,parent)

        if que.qsize() == 0:
            return "No path found"
