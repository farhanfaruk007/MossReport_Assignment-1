import heapq

def graph(file):
    main_graph = {}
    h_value = {}
    with open(file, 'r') as in_file:
        for line in in_file:
            list_line = line.split()
            loc = list_line[0]
            new_h_value = int(list_line[1])
            connected = []
            for i in range(2, len(list_line), 2):
                node = list_line[i]
                weight = int(list_line[i + 1])
                connected.append((node, weight))



            main_graph[loc] = connected
            h_value[loc] = new_h_value

    return main_graph, h_value

def a_star_algo(main_graph, start, goal, h_value):
    open_start_node_list= [(0 + h_value[start], start)]


    previous_city = {}
    g_score = {node: float('inf') for node in main_graph}
    g_score[start] = 0

    while open_start_node_list:
        f_score, current = heapq.heappop(open_start_node_list)
        if current == goal:
            final_path = []
            while current in previous_city:
                final_path.append(current)
                current = previous_city[current]
            final_path.append(start)
            return final_path[::-1]

        for neighbor, weight in main_graph[current]:
            new_g_score = g_score[current] + weight

            if new_g_score < g_score[neighbor]:
                previous_city[neighbor] = current
                g_score[neighbor] = new_g_score
                f_score = new_g_score + h_value[neighbor]
                heapq.heappush(open_start_node_list, (f_score, neighbor))

    return None

start = input("Enter the starting city: ")
goal = input("Enter the destination city: ")
x, y = graph("/content/sample_data/input.txt")
path = a_star_algo(x, start, goal, y)

if path:
    print(" -> ".join(path))


    distance = 0

    for i in range(len(path) - 1):
        current_location = path[i]
        next_location = path[i + 1]
        for adjacent_location, cost in x[current_location]:
            if adjacent_location == next_location:
                distance += cost
                break


    print(f"Total distance: {distance} km")
else:
    print("No Path Found")
