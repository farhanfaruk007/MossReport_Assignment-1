import heapq

def read(file):
    graph = {}
    heuristic = {}
    with open(file, 'r') as input_f:
        for x in input_f:
            parts = x.split()
            city = parts[0]
            heuristic[city] = int(parts[1])
            neighbors = {}
            for i in range(2, len(parts), 2):
                neighbor = parts[i]
                dist = int(parts[i + 1])
                neighbors[neighbor] = dist


            graph[city] = neighbors
    return graph, heuristic

def search(graph, heuristic, start, goal):
    list1 = []
    heapq.heappush(list1, (0, start))
    dict1 = {}
    g_score = {city: float('inf') for city in graph}
    g_score[start] = 0
    t_cost = {city: float('inf') for city in graph}
    t_cost[start] = heuristic[start]
    while list1:
        current_f, current = heapq.heappop(list1)
        if current == goal:
            path = []
            while current in dict1:
                path.append(current)
                current = dict1[current]
            path.append(start)
            path.reverse()
            return path, g_score[goal]
        for neighbor, dist in graph[current].items():
            temp_g_score = g_score[current] + dist
            if temp_g_score < g_score[neighbor]:


                dict1[neighbor] = current
                g_score[neighbor] = temp_g_score
                t_cost[neighbor] = g_score[neighbor] + heuristic[neighbor]
                heapq.heappush(list1, (t_cost[neighbor], neighbor))

    return None, float('inf')

def main():
    file = 'input.txt'
    graph, heuristic = read(file)
    start = input("Start node: ").strip()
    goal = input("Destination: ").strip()
    if start not in graph or goal not in graph:
        print("Start or destination node not found in the graph.")
        return


    path, dist = search(graph, heuristic, start, goal)
    if path:
        print("Path:", " -> ".join(path))
        print("Total distance:", dist, "km")
    else:
        print("NO PATH FOUND")

if __name__ == "__main__":
    main()
