inputfile = open('input1.txt', 'r')

dic = {}
dic1 = {}
nei = []

for i in inputfile:
    part = i.split()
    dic[part[0]] = int(part[1])
    nei = []
    for j in range(2, len(part), 2):
        nei.append((part[j], int(part[j + 1])))
    dic1[part[0]] = nei


def AsTarAlGorithm(st_n, stp_n):
    opening = set([st_n])
    closing = set()
    parents = {}
    parents[st_n] = st_n
    g = {}
    g[st_n] = 0




    while len(opening) > 0:
        n = None
        for q in opening:
            if n == None or (g[q] + dic[q] < g[n] + dic[n]):
                n = q
        if n == stp_n or dic1[n] == None:
            pass
        else:
            for (m, w) in dic1[n]:
                if m not in opening and m not in closing:
                    opening.add(m)
                    parents[m] = n
                    g[m] = g[n] + w
                else:
                    if g[m] > g[n] + w:


                        g[m] = g[n] + w
                        parents[m] = n
                        if m in closing:
                            closing.remove(m)
                            opening.add(m)


        if n == None:
            print('NO PATH FOUND')
            return None
        elif n == stp_n:
            path = []
            total_distance = g[n]
            while parents[n] != n:
                path.append(n)
                n = parents[n]
            path.append(st_n)
            path.reverse()
            print('Path: ' + ' -> '.join(path))
            print('Total distance:',total_distance,'km')

            return path
        opening.remove(n)
        closing.add(n)
    print('NO PATH FOUND')
    return None

AsTarAlGorithm('Arad', 'Bucharest')
