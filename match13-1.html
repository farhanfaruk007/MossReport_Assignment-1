# a* seearch


input=open('input_1.txt','r')
dicc={}
for i in input:
  info=i.split()
  start=info[0]
  h_value=int(info[1])
  sd=info[2:]
  # print(sd)
  dicc[start]={'h_value':h_value,'sd': {} }
  for i in range(0,len(sd),2):
    child,dis=sd[i],int(sd[i+1])
    dicc[start]['sd'][child]=dis
# print(dicc)


import heapq
def astar(st,des):
  path=[]
  path.append((0,st))
  total_dis={}
  for i in dicc:
    total_dis[i]=float('inf')
  total_dis[st]=0
  estimated={}
  for j in dicc:
    estimated[j]=float('inf')
  estimated[st]=dicc[st]['h_value']
  # print(estimated)

  out={}
  while len(path) > 0:
    x,y=heapq.heappop(path)
    if y == des:
      p=[]
      while y in out:
        p.append(y)
        y=out[y]
      p.append(st)
      p.reverse()
      # print(p)
      return p,total_dis[des]

    for cld,dis in dicc[y]['sd'].items():
      act=total_dis[y] + dis
      if act <total_dis[cld]:


        out[cld]=y
        total_dis[cld] = act
        estimated[cld]=total_dis[cld] +dicc[cld]['h_value']
        # print(estimated)
        heapq.heappush(path,(estimated[cld],cld))


  return None, float('inf')

start='Arad'
end='Bucharest'

a,b=astar(start,end)

if a is not None and len(a)>0:
  print('path:',end=" ")
  for i in range(len(a)):
    if i < len(a) -1:
      print(a[i],end="-->")
    else:
      print(a[i])

  print(f"total distnce {b} km")
else:
