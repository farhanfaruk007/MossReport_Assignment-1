from queue import PriorityQueue

def astar_algorithm(graph, start, goal):
    visited = PriorityQueue()
    visited.put((graph[start]['Heuristic'], start))
    weight = {start: 0}
    parent = {}
    explored = set()

    while not visited.empty():
        _, current_city = visited.get()

        if current_city == goal:
            path = []
            while current_city in parent:
                path.append(current_city)
                current_city = parent[current_city]
            path.append(start)
            path.reverse()
            return weight[goal], path

        explored.add(current_city)

        for neighbor, distance in graph[current_city]["Neighbors"].items():
            if neighbor in explored:
                continue

            new_weight = weight[current_city] + distance

            if neighbor not in weight or new_weight < weight[neighbor]:
                weight[neighbor] = new_weight
                f_n = new_weight + graph[neighbor]["Heuristic"]
                visited.put((f_n, neighbor))
                parent[neighbor] = current_city

    return None




graph={}

    info=line.split()

        neighbor_city=info[n_i]
        distance=float(info[n_i+1])
        neighbor_info[neighbor_city]=distance
        n_i+=2
    graph[city]={'Heuristic':heuristic,'Neighbor':neighbor_info}
