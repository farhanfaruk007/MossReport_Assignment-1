with open("Input file.txt", 'r') as file:
    dataset = file.readlines()

print(f"Original dataset- {dataset}")

def find_heuristic(lines):
    h_dict = {}
    for each_line in lines:
        all_line = each_line.strip().split()
        h_dict[all_line[0]] = int(all_line[1])
    return h_dict
heuristic = find_heuristic(dataset)

print(f'Converted dataset according to its heuristic- {heuristic}')

def nextdoor(val, lines):
    graph = {}
    adj_graph = {}

    for ind in lines:
        line = ind.strip().split()
        len_line = len(line)

        for x in range(2, len_line, 2):
            adj_graph[line[x]] = int(line[x + 1])
        graph[line[0]] = adj_graph
        adj_graph = {}

    return graph[val]





def a_star_algo(start_n, stop_n):
    openlist = [start_n]
    closedlist = []
    g = {}
    parentlist = {}
    finaldist = {}

    g[start_n] = 0
    parentlist[start_n] = start_n
    finaldist[start_n] = 0



    while len(openlist) > 0:
        nd = None

        for red in openlist:
            if nd is None or g[red] + heuristic[red] < g[nd] + heuristic[nd]:
                nd = red

        if nd is None:
            print('NO PATH FOUND!')
            return None

        if nd == stop_n:
            paath = []
            while parentlist[nd] != nd:
                paath.append(nd)
                nd = parentlist[nd]
            paath.append(start_n)
            paath.reverse()

            print('Path: {}'.format(paath))
            curr_dist = finaldist[stop_n]
            print('Total distance: {}'.format(curr_dist),'km')
            return None

        for (neighbor, weight) in nextdoor(nd, lines).items():

            if neighbor not in openlist and neighbor not in closedlist:
                openlist.append(neighbor)
                parentlist[neighbor] = nd
                g[neighbor] = g[nd] + weight
                finaldist[neighbor] = finaldist[nd] + weight
            else:
                if g[neighbor] > g[nd] + weight:
                    g[neighbor] = g[nd] + weight
                    parentlist[neighbor] = nd
                    finaldist[neighbor] = finaldist[nd] + weight


                    if neighbor in closedlist:
                        closedlist.remove(neighbor)
                        openlist.append(neighbor)

        openlist.remove(nd)
        closedlist.append(nd)

    print('NO PATH FOUND!')
    return None

start_node = input('Enter start state- ')
goal_node = input('Enter goal state- ')
