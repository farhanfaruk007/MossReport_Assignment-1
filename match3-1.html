with open('Input file.txt', 'r') as file:
    content = file.readlines()
Keyword = {'Arad': 'A', 'Bucharest': 'Z', 'Craiova': 'S', 'Eforie': 'T', 'Fagaras': 'O', 'Dobreta': 'V',
           'Hirsova': 'N', 'lasi': 'Q', 'Lugoj': 'G', 'Mehadia': 'L', 'Neamt': 'F', 'Oradea': 'B', 'Pitesti': 'P', 'RimnicuVilcea': 'R',
           'Timisoara': 'C', 'Urziceni': 'D', 'Vaslui': 'H', 'Zerind': 'E','Giurgiu':'M','Sibiu':'U'}


Heuristic_val={}
graph={}
temp={}

for line in content:
    line = line.strip()
    line = line.split(" ")
    for j in range(2,len(line)-1):
      if j%2==0:
        temp[Keyword[line[j]]]=line[j+1]
    graph[Keyword[line[0]]]=temp
    temp={}
    Heuristic_val[Keyword[line[0]]]=line[1]

Q={}
visited=[]
Final_Temp=[]
Temp=""
first_key="A"
dest=Keyword["Bucharest"]
dist=0
Total_dist=0
while Temp!="Z":
  if Temp=="":
    Q[first_key]=int(Heuristic_val[first_key])+dist
    Temp=first_key
    Final_Temp.append(Temp)
    visited.append(Temp)
    Q.popitem()[0]
  else:
    for i in graph[Temp]:
      if i not in Q and  i not in visited:
        Q[i]=int(Heuristic_val[i])+int(graph[Temp][i])+dist
        visited.append(i)
    Total_dist+=dist
    Q = dict(sorted(Q.items(), key=lambda item: item[1]))
    dist=graph[Temp][list(Q.keys())[0]]
    dist=int(dist)
    Temp = list(Q.keys())[0]
    Q.pop(Temp)
    Final_Temp.append(Temp)

print("Path: ",end="")
count=1
for final in Final_Temp:
  if dest not in Final_Temp:
    print("NO PATH FOUND")
  else:
    for keyy,valuee in Keyword.items():
      if valuee==final and count!=len(Final_Temp):
        print(keyy,end="-->")
        count+=1
      elif valuee==final and count==len(Final_Temp):
        print(keyy)

print("Total Distance: ",Total_dist+int(graph[Temp][visited[-2]]),"KM")
