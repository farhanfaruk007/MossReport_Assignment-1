def astarSearch(graph, start, goal, heur1st1cL1st):
    notVisited = [(0, start)]
    visited = set()
    parent = {}
    cost = {}
    parent[start]= None
    cost[start]= 0

    while notVisited:
         m1n_1ndex= 0
         for l in range(1, len(notVisited)):
            if  notVisited[l][0] < notVisited[m1n_1ndex][0]:
              m1n_1ndex= l



         curr_cost, curr = notVisited.pop(m1n_1ndex)

         if curr in visited:
              continue

         visited.add(curr)

         if curr == goal:
           path = []
           total_cost = cost[curr]
           while curr != None:
              path.append(curr)
              curr=parent[curr]
           path.reverse()
           return path,total_cost

         for ne1ghbor , path_cost in graph.get(curr, []):
           new_cst = cost[curr]+ path_cost
           if ne1ghbor not in cost or new_cst < cost[ne1ghbor]:
             cost[ne1ghbor] = new_cst
             priority = new_cst + heur1st1cL1st.get(ne1ghbor, float('inf'))
             notVisited.append((priority, ne1ghbor))
             parent[ne1ghbor] = curr
    return None , float('inf')
