import sys
import heapq


startNode = input("Start node: ")
goalNode = input("Destination: ")


sys.stdin = open("input.txt", "r")

closest_node = {}
city_map = {}
heuristic_val = {}
while True:
    try:
        inp = input().split(" ")
        MC = inp[0]
        if MC not in city_map:
            city_map[MC] = {}

        for i in range(2, len(inp), 2):
            heuristic_val[inp[0]] = int(inp[1])
            if i + 1 < len(inp):
                CT = inp[i]
                DIS = int(inp[i + 1])
                closest_node[CT] = DIS

                city_map[inp[0]][CT] = DIS

    except EOFError:
        break





def AStar(graph, heuristic_val, startNode, goalNode):
    pq_set = []
    heapq.heappush(pq_set, (heuristic_val[startNode], startNode))
    gnCost = {}
    fnCost = {}
    backtrack_path = {}

    for i in graph:
        gnCost[i] = float('inf')
        fnCost[i] = float('inf')
    gnCost[startNode] = 0
    fnCost[startNode] = heuristic_val[startNode]

    while pq_set:
        current_fnCost , currentNode = heapq.heappop(pq_set)

        if currentNode == goalNode:
            actual_path = []
            while currentNode in backtrack_path:
                actual_path.append(currentNode)
                currentNode = backtrack_path[currentNode]
            actual_path.append(startNode)
            actual_path.reverse()
            return (actual_path , gnCost[goalNode])
        else:
            for closest_node,dist in graph[currentNode].items() :


                total_gnCost = gnCost[currentNode] + dist

                if total_gnCost < gnCost[closest_node]:
                    gnCost[closest_node] = total_gnCost
                    fnCost[closest_node] = gnCost[closest_node] + heuristic_val[closest_node]
                    backtrack_path[closest_node] = currentNode

                    heapq.heappush(pq_set, (fnCost[closest_node], closest_node))

    return "NO PATH FOUND"



a_star_path , total_cost = AStar(city_map, heuristic_val, startNode, goalNode)

print("Path:" ,end = "")
for path in range(len(a_star_path)):

    if len(a_star_path) == path+1:
        print(a_star_path[path])
    else:
        print(f" {a_star_path[path]}", end = " -> ")

print(f"Total Distance: {total_cost} km")
