import heapq

def cities_graph(file):

  graph = {}
  heuristics = {}

  with open(file, 'r') as x:
    for i in x:
      index = i.strip().split()


      city = index[0]
      heuristic = int(index[1])
      heuristics[city] = heuristic
      neighbors = index[2: ]
      graph[city] = {}


      for j in range (0, len(neighbors), 2):
        next = neighbors[j]
        distance = int(neighbors[j+1])
        graph[city][next] = distance


  return graph, heuristics


def a_star_search(graph, heuristics, start, destination):
  priority_queue = []
  heapq.heappush(priority_queue, (heuristics[start], 0, start, [start]))

  visited = {}

  while priority_queue:
    f, g, current, path = heapq.heappop(priority_queue)         # g = cumulative dist. from start to current node

    if current == destination:
      return path, g

    if current in visited and visited[current] <= g:
      continue

    visited[current] = g

    for next, distance in graph[current].items():
      new_g = g + distance
      new_f = new_g + heuristics[next]
      heapq.heappush(priority_queue, (new_f, new_g, next, path + [next]))

  return None, None


file = 'Input file.txt'
graph, heuristics = cities_graph(file)


start = input("Enter start node: ").strip()
destination = input("Enter destination node: ").strip()


path, distance = a_star_search(graph, heuristics, start, destination)


if path:
  print("Path:", " -> ".join(path))
  print("Total Distance:", distance, "km")

else:
  print("NO PATH FOUND")
