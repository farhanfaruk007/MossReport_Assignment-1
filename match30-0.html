import heapq



def astar_algorithm(start_node, target_node, graph_structure, heuristic_map):
    priority_queue = []
    heapq.heappush(priority_queue, (0, start_node))
    node_traceback = {start_node: None}
    cost_so_far = {start_node: 0}

    while priority_queue:
        current_vertex = heapq.heappop(priority_queue)[1]
        if current_vertex == target_node:
            break
        for adjacent_vertex in graph_structure[current_vertex]:
            new_cost = cost_so_far[current_vertex] + graph_structure[current_vertex][adjacent_vertex]
            if adjacent_vertex not in cost_so_far or new_cost < cost_so_far[adjacent_vertex]:
                cost_so_far[adjacent_vertex] = new_cost
                estimated_cost = new_cost + heuristic_map[adjacent_vertex]
                heapq.heappush(priority_queue, (estimated_cost, adjacent_vertex))
                node_traceback[adjacent_vertex] = current_vertex
    return node_traceback, cost_so_far



def create_graph(input_file):
    graph_structure = {}
    temp_edges = {}
    for line in input_file:
        parts = line.strip().split()
        for i in range(2, len(parts), 2):
            temp_edges[parts[i]] = int(parts[i + 1])
        graph_structure[parts[0]] = temp_edges
        temp_edges = {}
    return graph_structure
