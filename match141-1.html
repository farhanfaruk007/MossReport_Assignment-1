# Reading input from file and converting to dictionary


h_table = {}
graph = {}
with open("Input.txt") as file:
    for line in file:
        line = line.strip().split(" ")
        h_table[line[0]] = int(line[1])
        temp = {}
        for i in range(2, len(line), 2):
            temp[line[i]] = int(line[i + 1])
        graph[line[0]] = temp

# Defining Start and Destination
start = input("Start node: ").capitalize()
destination = input("Destination: ").capitalize()

def a_star(start, destination, graph, h_table):
    # Creating Priority Queue (Fringe) using heapdict
    import heapdict

    q = heapdict.heapdict()
    parent = {i: None for i in h_table.keys()}
    p_cost = {i: float("inf") for i in h_table.keys()}


    q[start] = h_table[start]
    p_cost[start] = 0

    while len(q) != 0:
        n = q.popitem()

        if n[0] == destination:
            current = destination
            path = [destination]

            while parent[current] != None:
                path.insert(0, parent[current])
                current = parent[current]

            print("Path: " + " -> ".join(path))
            print("Total distance:", p_cost[destination], "km")
            return "Success"

        else: 
            for successor in graph[n[0]]:
                n_c = graph[n[0]][successor]


                g_c = p_cost[n[0]] + n_c
                
                if g_c < p_cost[successor]:
                    p_cost[successor] = g_c
                    parent[successor] = n[0]
                    f_n = g_c + h_table[successor]
                    q[successor] = f_n
                    
    print("NO PATH FOUND")
    return "Error"


a_star(start, destination, graph, h_table)
