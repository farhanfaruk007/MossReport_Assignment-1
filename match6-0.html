def original_g_dict(x):
    if x in original_G_dict:
        return original_G_dict[x]

        return None

def a_Start_sear(starting_node, ending_node):
    store1 = set()
    store1.add(starting_node)
    store2 = set()
    dict1 = {}
    total_cost = {}
    total_cost[starting_node] = 0
    dict1[starting_node] = starting_node
    while len(store1)>0:
        u_val=None

        for j in store1:
            if u_val is None or total_cost[j] + hurestic_value[j] < total_cost[u_val] + hurestic_value[u_val]:
                u_val = j
        if u_val == ending_node or original_G_dict[u_val] == None:
            pass

        else:
            for (nextValue, Lod) in original_g_dict(u_val):
                    if nextValue not in store1 and nextValue not in store2:
                        store1.add(nextValue)
                        dict1[nextValue] = u_val
                        total_cost[nextValue] = total_cost[u_val] + Lod
                        path_distance = total_cost[nextValue]
                    else:
                        if total_cost[nextValue] > total_cost[u_val] + Lod:
                            total_cost[nextValue] = total_cost[u_val] + Lod
                            dict1[nextValue] = u_val

                            if nextValue in store2:
                                store2.remove(nextValue)
                                store1.add(nextValue)
                        path_distance = total_cost[nextValue]


        if u_val == None:
            print('No path found')
            return None

        if u_val == ending_node:
            tester_list=[]

            while dict1[u_val] != u_val:
                tester_list.append(u_val)
                u_val = dict1[u_val]
            tester_list.append(starting_node)
            tester_list.reverse()

            print(*tester_list, sep =" ->")
            print("Total Distance:", path_distance,"km")
            return None
        store1.remove(u_val)
        store2.add(u_val)
    print('No path found')
    return None
