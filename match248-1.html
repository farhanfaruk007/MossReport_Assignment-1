input_file = '/content/drive/MyDrive/CSE422/input.txt'

import heapq

def parse(file):
  graph = {}
  heu_value = {}

  with open(file, 'r') as inp:
    for line in inp:
      portion = line.strip().split()
      if len(portion) < 2:
        continue

      head = portion[0]
      value = int(portion[1])
      heu_value[head] = value
      kids = []

      for i in range(2, len(portion), 2):
        child = portion[i]
        gap = int(portion[i + 1])
        kids.append((child,gap))

      graph[head] = kids

    return graph, heu_value

def s_search(graph,heu_value,start,destination):
  n_set = []


  heapq.heappush(n_set, (0 + heu_value[start], 0, start, [start]))
  checked = set()

  while n_set:
    est_cost, g_cost, c_node, path = heapq.heappop(n_set)

    if c_node == destination:
      return path,g_cost

    if c_node in checked:
      continue
    checked.add(c_node)

    for child, gap in graph.get(c_node, []):
      if child not in checked:
        n_g_cost = g_cost + gap
        f_cost = n_g_cost + heu_value[child]
        heapq.heappush(n_set, (f_cost, n_g_cost,child, path + [child]))

  return None, None

def main():
  graph, heu_value = parse(input_file)



  start = input("Start node: ").strip()
  destination = input("Destination: ").strip()

  if start not in graph or destination not in graph:
    print("NO PATH FOUND")
    return

  path, t_distance = s_search(graph,heu_value,start,destination)

  if path:
    print("Path: " + " ->".join(path) + "\n")
    print("Total distance: " + str(t_distance) + "km\n")

  else:
    print("NO PATH FOUND\n")

main()
