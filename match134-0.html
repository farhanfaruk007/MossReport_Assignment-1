import heapq

class TravelMap:
    def __init__(self):
        self.superman = {}

    def rasta(self, cityA, cityB, distance=0):
        if cityA not in self.superman:
            self.superman[cityA] = []

        if cityB not in self.superman:
            self.superman[cityB] = []

        if (cityA, distance) not in self.superman[cityB]:
            self.superman[cityB].append((cityA, distance))

        if (cityB, distance) not in self.superman[cityA]:
            self.superman[cityA].append((cityB, distance))



    def pashapashi_cities(self, city):

    def showMap(self):

def heuristic_er_jonno(city, estimates):
    for location, value in estimates:
        if location == city:
            return value
    return 0

def searchRoute(travelMap, startCity, endCity, estimates):
    pendingCities = [(0, startCity)]
    visitedCities = set()
    dis = {startCity: 0}
    routeHistory = {}

    while pendingCities:
        _, activeCity = heapq.heappop(pendingCities)

        if activeCity == endCity:

            route = []
            total_distance = dis[activeCity]
            while activeCity in routeHistory:
                route.append(activeCity)
                activeCity = routeHistory[activeCity]
            route.append(startCity)

            return list(reversed(route)), total_distance

        visitedCities.add(activeCity)

        for shater_City, travel_dis in travelMap.pashapashi_cities(activeCity):
            total = dis[activeCity] + travel_dis

            if shater_City in visitedCities or total >= dis.get(shater_City, float('inf')):
                continue

            if total < dis.get(shater_City, float('inf')):
                routeHistory[shater_City] = activeCity
                dis[shater_City] = total
                estimated_full_dis = total + heuristic_er_jonno(shater_City, estimates)
                heapq.heappush(pendingCities, (estimated_full_dis, shater_City))

    return None, 0
