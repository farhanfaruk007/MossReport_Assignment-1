import heapq


f_in = open("/content/Input file.txt", "r")

hVal = {}
city_graph = {}
lines = f_in.readlines()


for j in lines:
    city_l = j.split()
    city = city_l[0]
    heu_city = int(city_l[1])
    hVal[city] = heu_city

    neigh_nodes = {}

    j = 2
    while j < len(city_l):
        nCity = city_l[j]
        distance = int(city_l[j + 1])
        neigh_nodes[nCity] = distance
        j += 2

    city_graph[city] = neigh_nodes



f_in.close()



def AStarSearch(graph, heu_value, sNode, gNode):
    prioQ = []
    heapq.heappush(prioQ, (heu_value[sNode], sNode))
    goalNodeCost = {}
    funcCost = {}
    traceback = {}

    for i in graph:
        goalNodeCost[i] = float('inf')
        funcCost[i] = float('inf')

    goalNodeCost[sNode] = 0
    funcCost[sNode] = heu_value[sNode]

    while prioQ:
        current_funcCost, currentNode = heapq.heappop(prioQ)

        if currentNode == gNode:
            act_path = []
            while currentNode in traceback:
                act_path.append(currentNode)
                currentNode = traceback[currentNode]

            act_path.append(sNode)
            act_path.reverse()
            return act_path, goalNodeCost[gNode]

        for close_node, d in graph[currentNode].items():


            t_goalNodeCost = goalNodeCost[currentNode] + d

            if t_goalNodeCost < goalNodeCost[close_node]:
                goalNodeCost[close_node] = t_goalNodeCost
                funcCost[close_node] = goalNodeCost[close_node] + heu_value[close_node]
                traceback[close_node] = currentNode

                heapq.heappush(prioQ, (funcCost[close_node], close_node))

    return None


startNode = input("Start: ")
goalNode = input("Destination: ")

result = AStarSearch(city_graph, hVal, startNode, goalNode)

if result:
    p, t_d = result
    print(f"Path: {' -> '.join(p)}")
    print(f"Total Distance: {t_d} km")
else:
    print("NO PATH FOUND")
