import heapq
input_info = []
heuristics = {}
neighbor_distance = {}
inp_file = open("input1.txt", "r")


for lines in inp_file:
  info = lines.strip().split()
  input_info.append(info)
  city =info[0]
  heuristics[city] = int(info[1])

  for i in range(2, len(info)-1, 2):
    neighbor_city = info[i]
    distance = int(info[i+1])
    if city not in neighbor_distance:
        neighbor_distance[city] = []
    neighbor_distance[city].append((neighbor_city, distance))


inp_file.close()

def A_search(start, goal, neighbor_distance, heuristics):
  queue = []
  explored = set()
  heapq.heappush(queue, (heuristics[start], 0, start, []))
  if queue==False:
    return None,float('inf')



  while queue:
    state, cost, current, path = heapq.heappop(queue)
    if current in explored:
      continue
    path = path + [current]

    if current == goal:
      return path, cost
    explored.add(current)


    neighbors = neighbor_distance[current]
    i = 0
    while i < len(neighbors):
      neighbor, distance = neighbors[i]
      if neighbor not in explored:
        new_cost = cost + distance
        estimated_cost = new_cost + heuristics[neighbor]
        heapq.heappush(queue, (estimated_cost, new_cost, neighbor, path))
      i += 1
start = input('Start node: ')
goal = input('Destination: ')
path, cost = A_search(start, goal, neighbor_distance, heuristics)

if path==False:
  print("NO PATH FOUND.")
else:
  selected_cities =""
  for cities in path:
    selected_cities += cities + " -> "
  print(f"Path: {selected_cities[:-3]}")
  print(f"Total Cost: {cost} km")
