import heapq


class priorityQueue:
    def __init__(self):
        self.cities = []
    def push(self, city, cost):
        heapq.heappush(self.cities,(cost, city))
    def pop(self):
        return heapq.heappop(self.cities)[1]
    def isEmpty(self):
        if self.cities == []:
            return True
        else:
            return False
    def check(self):
        print(self.cities)

class cityNode:

def heuristic(node, values):
