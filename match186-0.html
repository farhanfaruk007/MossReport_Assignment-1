import heapq
from collections import defaultdict

def take_input_text(file):
    input_info = []
    neighbor_distance = defaultdict(list)
    heuristics = {}

    with open("file.txt", 'r') as input_lines:


      for line in input_lines:
        details = line.strip().split()
        input_info.append(details)
        city = details[0]
        heuristics[city] = int(details[1])

        for i in range(2, len(details)-1, 2):
          neighbor_city = details[i]
          distance = int(details[i+1])
          neighbor_distance[city].append((neighbor_city, distance))



    return input_info, neighbor_distance, heuristics


def A_search(start, goal, neighbor_distance, heuristics):
    p_queue = []
    exlored_city = set()
    heapq.heappush(p_queue, (heuristics[start], 0, start, []))



    while p_queue:
      state, cost, current_city, path = heapq.heappop(p_queue)

      if current_city in exlored_city:
        continue

      path = path + [current_city]

      if current_city == goal:
        return path, cost

      exlored_city.add(current_city)

      for neighbor, distance in neighbor_distance[current_city]:
        if neighbor not in exlored_city:
          new_cost = cost + distance
          estimated_cost = new_cost + heuristics[neighbor]
          heapq.heappush(p_queue, (estimated_cost, new_cost, neighbor, path))


    return None, float('inf')


#Inputs
file_path = '/content/Input_file.txt'
input_info, neighbor_distance, heuristics = take_input_text(file_path)
start = input('Start node: ')
goal = input('Destination: ')
path, cost = A_search(start, goal, neighbor_distance, heuristics)

if path is None:
  print("NO PATH FOUND.")
else:
  p =""
  for city in path:
    p += city + " -> "
  print(f"Path: {p[:-3]}")
  print(f"Total Cost: {cost} km")
