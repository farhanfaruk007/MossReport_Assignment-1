import heapq

def A(out, start, end):
    pq = [(0, start)]
    truecost = {start: 0}
    e = {start: out[start]['h']}
    fpath = {}

    while pq:
        current_cost, current_city = heapq.heappop(pq)
        if current_city == end:
            path = []
            while current_city in fpath:
                path.append(current_city)
                current_city = fpath[current_city]
            path.append(start)


            return path[::-1], truecost[end]

        for neighbor, distance in out[current_city]['children'].items():
            new_cost = truecost[current_city] + distance

            if neighbor not in truecost or new_cost < truecost[neighbor]:


                fpath[neighbor] = current_city
                truecost[neighbor] = new_cost
                e[neighbor] = new_cost + out[neighbor]['h']
                heapq.heappush(pq, (e[neighbor], neighbor))

    return None, float('inf')

start ='Arad'
