def inp_graph(input_file):
    graph = {}
    with open(input_file, "r") as file:
        for line in file:
            info = line.split()
            city = info[0]
            heuristic = int(info[1])
            neighbor = {}

            for i in range(2, len(info), 2):
                edge = info[i]


                distance = int(info[i + 1])
                neighbor[edge] = distance
            graph[city] = {"h": heuristic, "e": neighbor}
    return graph

# A* algorithm
def a_star(start, goal, graph):
    open_path = {start}
    closed_path = set()
    path_cost = {start: 0}
    final_cost = {start: graph[start]["h"]}
    track_p = {}

    while open_path:
        current_node = min(open_path, key=lambda node: final_cost[node])

        if current_node == goal:
            path_c = [current_node]
            while current_node in track_p:
                current_node = track_p[current_node]
                path_c.append(current_node)
            path_c.reverse()
            return (final_cost[goal], path_c)
        open_path.remove(current_node)
        closed_path.add(current_node)

        for edge, d in graph[current_node]["e"].items():
            if edge in closed_path:
                continue
            temp_cost = path_cost[current_node] + d
            if edge not in open_path:
                open_path.add(edge)
            elif temp_cost >= path_cost[edge]:
                continue
            track_p[edge] = current_node
            path_cost[edge] = temp_cost
            final_cost[edge] = path_cost[edge] + graph[edge]["h"]
    return None

def main():
    graph = inp_graph("Input file.txt")


    start = input("Start node: ")
    goal = input("Destination: ")
    result = a_star(start, goal, graph)
    if result is None:
        print("NO PATH FOUND")
    else:
        distance, path = result
        print("Path:", "-> ".join(path))
        print("Total distance:", distance, "km")

if __name__ == "__main__":
    main()
