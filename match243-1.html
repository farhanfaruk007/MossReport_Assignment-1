import heapq


def my_info(file_path):
    graph = {}
    heuristics = {}

    with open("/content/lab01_cse422.txt", 'r') as file:
        for i in file:
            parts = i.split()
            city = parts[0]
            heuristic = int(parts[1])
            heuristics[city] = heuristic

            neighbors = {}
            for k in range(2, len(parts), 2):
                neighbor = parts[k]
                distance = int(parts[k + 1])
                neighbors[neighbor] = distance

            graph[city] = neighbors

    return graph, heuristics


def a_star_search(graph, heuristics, start, goal):

    new = []
    heapq.heappush(new, (0 + heuristics[start], start))


    g_scores = {city: float('inf') for city in graph}
    g_scores[start] = 0


    came_from = {}



    while new:
        current_f_score, current_city = heapq.heappop(new)

        if current_city == goal:
            return reconstruct_path(came_from, current_city), g_scores[goal]

        for neighbor, distance in graph[current_city].items():
            tentative_g_score = g_scores[current_city] + distance

            if tentative_g_score < g_scores[neighbor]:
                came_from[neighbor] = current_city
                g_scores[neighbor] = tentative_g_score
                f_score = tentative_g_score + heuristics[neighbor]
                heapq.heappush(new, (f_score, neighbor))

    return None, float('inf')




def reconstruct_path(came_from, current):
    path = []
    while current in came_from:
        path.append(current)
        current = came_from[current]
    path.append(current)
    return path[::-1]

def main():
    file_path = 'input.txt'

    graph, heuristics = my_info(file_path)

    start = input("Enter the start city (e.g., Arad): ")
    goal = input("Enter the goal city (e.g., Bucharest): ")

    path, total_distance = a_star_search(graph, heuristics, start, goal)

    if path:
        print(f"Path: {' -> '.join(path)}")
        print(f"Total distance: {total_distance} km")
    else:
        print("NO PATH FOUND")

if __name__ == "__main__":
    main()
