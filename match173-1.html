from heapq import heappop, heappush

def a_star_search(graph, start, goal, heuristics):
    open_set, evaluated_nodes = [(heuristics[start], start)], set()
    path = []

    while open_set:
        current_cost, node = heappop(open_set)
        path.append(node)

        if node == goal:
            return [current_cost, path]



        if node in evaluated_nodes:
            continue
        evaluated_nodes.add(node)

        current_cost -= heuristics[node]

        for neighbor, edge_cost in graph[node]:
            if neighbor in evaluated_nodes:
                continue

            if neighbor == goal:
                all_routes_to_goal[node] = (neighbor, current_cost + edge_cost)

            estimated_cost = current_cost + edge_cost + heuristics[neighbor]
            heappush(open_set, (estimated_cost, neighbor))

    return -1
input_file = "input"
graph, heuristics = {}, {}
all_routes_to_goal = {}



with open(input_file, 'r') as file:
    for line in file:
        parts = line.split()
        neighbors = []
        for i in range(2, len(parts), 2):


            if i + 1 < len(parts):
                neighbors.append((parts[i], int(parts[i + 1])))

        graph[parts[0]] = neighbors
        heuristics[parts[0]] = int(parts[1])

start_node = input("Start node: ")
goal_node = input("Destination: ")

result = a_star_search(graph, start_node, goal_node, heuristics)

if result != -1:
    total_distance = result[0]
    path = result[1]

    if heuristics[goal_node] != 0:
        total_distance -= heuristics[goal_node]

    lowest_path_cost = min(all_routes_to_goal.values(), default=None)
    for node, route in all_routes_to_goal.items():
        if route != lowest_path_cost and node in path:
            path.remove(node)

    print('Path: ', ' -> '.join(path))
    print(f"Total Distance: {total_distance} km")
else:
    print('NO PATH FOUND')
