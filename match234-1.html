import heapq

def ASTAR(start, goal):
    visited_path= []
    visited_path.append((0,start))

    parent= {}
    dist = {}
    estimated_dist = {}

    for i in all_info:
      dist[i] = float('inf')
    dist[start] = 0


    for j in all_info:
      estimated_dist[j] = float('inf')
    estimated_dist[start] = all_info[start]['heuristic']

    while len(visited_path) > 0:
        d,country = heapq.heappop(visited_path)

        if country == goal:
            path = []
            while country in parent:
                path.append(country)
                country = parent[country]
            path.append(start)
            path.reverse()


            return path, dist[goal]

        for child, dt in all_info[country]['neighbours'].items():

            total_dist = dist[country] + dt
            if total_dist < dist[child]:


                parent[child] = country
                dist[child] = total_dist
                estimated_dist[child] = total_dist + all_info[child]['heuristic']
                heapq.heappush(visited_path, (estimated_dist[child],child))

    return None, float('inf')



all_info = {}

with open('/content/input.txt', 'r') as inputfile:

    for i in inputfile:

        list1 = i.split()

        country = list1[0]
        heuristic = int(list1[1])
        neighbours = list1[2:]

        all_info[country] = {'heuristic': heuristic, 'neighbours': {}}

        for j in range(0, len(neighbours), 2):
            all_info[country]['neighbours'][neighbours[j]] = int(neighbours[j + 1])

start = input("start node:")
goal = input("destination:")

path, distance = ASTAR(start,goal)

if  path is not None:


    print("Path:", end=" ")
    for i in range(len(path)):
        if i < len(path) - 1:
            print(path[i], end=" -> ")
        else:
            print(path[i])

    print("total distance:", distance, "km")
else:
    print("no path found")
