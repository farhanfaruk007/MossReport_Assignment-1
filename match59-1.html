import heapq

def read_input(inp) :
  graph = {}
  heuristics = {}

  with open(inp, 'r') as inp :
    for line in inp:
      temp = line.split()


      parent_city = temp[0]
      heuristic = int(temp[1])
      childs = temp[2:]
      graph[parent_city] = {}

      for i in range(0, len(childs), 2) :
        child = childs[i]
        dis = int(childs[i+1])
        graph[parent_city][child] = dis

      heuristics[parent_city] = heuristic

  return graph, heuristics

def A_star(graph, heuristics, start_state, goal_state) :
  lis = []
  heapq.heappush(lis, (0, start_state))
  visited = {}
  g_cost = {start_state: 0}
  f_cost = {start_state: heuristics[start_state]}

  while lis :
    current = heapq.heappop(lis)[1]

    if current == goal_state :
      path = []
      while current in visited :
        path.append(current)
        current = visited[current]
      path.append(start_state)
      path.reverse()

      return path, g_cost[goal_state]

    for child in graph[current] :
      tentative_g_cost = g_cost[current] + graph[current][child]

      if child not in g_cost or tentative_g_cost < g_cost[child] :
        visited[child] = current
        g_cost[child] = tentative_g_cost
        f_cost[child] = tentative_g_cost + heuristics[child]
        heapq.heappush(lis, (f_cost[child], child))

  return None, None

def main() :

    inp = '/content/Input file.txt'
    graph, heuristics = read_input(inp)

    start_state = input("Start node: ").strip()
    goal_state = input("Destination: ").strip()

    path, dis = A_star(graph, heuristics, start_state, goal_state)

    if path :
        print(f"Path: {' -> '.join(path)}")
        print(f"Total distance: {dis} km")
    else :
        print("NO PATH FOUND")

if __name__ == "__main__" :
    main()
