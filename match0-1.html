class PriorQue:
    def __init__ (self, sz):
      self.que=[]
      self.prior=[]
      self.path=[]


    def sortQue(self):
      for m in range(len(self.que)):
        for n in range(len(self.que)):
          if self.prior[n]>self.prior[m] and (self.prior[n]!=0 and self.prior[m]!=0):
            nd=self.que[m]
            val=self.prior[m]
            path=self.path[m]
            self.que[m]=self.que[n]
            self.que[n]=nd
            self.prior[m]=self.prior[n]
            self.prior[n]=val
            self.path[m]=self.path[n]
            self.path[n]=path


    def enQue(self,nd,cst,path):
       self.que.append(nd)
       self.prior.append(cst)
       self.path.append(path)
       self.sortQue()



    def deQue(self):
       elem=(self.que[0],self.prior[0],self.path[0])
       for k in range(1, len(self.que)):
         self.que[k-1]=self.que[k]
         self.prior[k-1]=self.prior[k]
         self.path[k-1]=self.path[k]
       del self.que[-1]
       del self.prior[-1]
       del self.path[-1]
       return elem



    def print_que(self):
      lent = len(self.que)
      for k in range(lent):
        print(self.que[k], end=' ')
      print()
      for l in range(lent):
        print(self.prior[l], end=' ')
      print()
      for i in range(lent):
        print(self.path[i], end=' ')
      print()


    def is_Emp(self):
      return self.que[0]==0
