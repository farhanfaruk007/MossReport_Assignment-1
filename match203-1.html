import heapq

n_lst = {}

with open('/content/sample_data/input.txt', 'r') as inputt:
    for i in inputt:
        lstt = i.split()

        d = lstt[2:]
        v = int(lstt[1])
        n = lstt[0]

        n_lst[n] = {'hv': v, 'fd': {}}

        for i in range(0, len(d), 2):
            chi, msh = d[i], int(d[i + 1])
            n_lst[n]['fd'][chi] = msh

start = input("Start node: ")
desti = input("Destination: ")

##############################

a, b = new(start, desti)

if a is not None and len(a) > 0:


    print("Path:", end=" ")
    for x in range(len(a)):
        if x < len(a) - 1:
            print(a[x], end=" -> ")
        else:
            print(a[x])

    print("Total distance:", b, "km")
else:
    print("NO PATH FOUND")

###############################

def new(mtt, mmm):
    trk = []
    trk.append((0, mtt))

    actu = {}
    for i in n_lst:
      actu[i] = float('inf')
    actu[mtt] = 0

    m = {}
    for k in n_lst:
      m[k] = float('inf')

    m[mtt] = n_lst[mtt]['hv']
    pt = {}



    while len(trk) > 0:
        x, cn = heapq.heappop(trk)

        if cn == mmm:
            q = []
            while cn in pt:
                q.append(cn)
                cn = pt[cn]

            q.append(mtt)
            q.reverse()

            return q, actu[mmm]

        for l, s in n_lst[cn]['fd'].items():
            act = actu[cn] + s

            if act < actu[l]:
                pt[l] = cn
                actu[l] = act
                m[l] = actu[l] + n_lst[l]['hv']
                heapq.heappush(trk, (m[l], l))

    return None, float('inf')
