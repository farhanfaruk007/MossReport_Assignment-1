import heapq #import priority queue for explor path

file_input = open('/content/input.txt')

g = {} #use dictionary key=city val=child
heuristic = {} #use dictionary key=city val=huristic val



def graph_creation(nodes): #graph
    parent_node = nodes[0] #current node
    g[parent_node] = {}
    for i in range(2, len(nodes), 2):
        children = nodes[i] #clild node
        weight = int(nodes[i + 1]) #cost
        g[parent_node][children] = weight

def A_star(begin, destination):
    A_h = [] #push start node
    heapq.heappush(A_h, (int(heuristic[begin]), 0, begin, [begin])) #sum of current path cost and huristic

    vist = [] #keep track of visited nodes
    while A_h:
        _, weight, city, path = heapq.heappop(A_h) #pop lowest estimated cost
        if city in vist:
            continue
        vist.append(city)
        if city == destination:
            return weight, path
        for children, cost in g[city].items():
            total_weight = weight + cost
            p_value = total_weight + heuristic[children]
            heapq.heappush(A_h, (p_value, total_weight, children, path + [children]))
    return None

def heuristic_up(val):
    heuristic[val[0]] = int(val[1]) #update heu dictionary with current node and heuristic value



r = file_input.readlines() # read input
for j in r:
    infos = j.split()
    heuristic_up(infos)
    graph_creation(infos)

begin = input("Enter start :") #ask for start
end = input("Enter End :") #ask for goal node

answer = A_star(begin, end) #call a star algo


output = open('output.txt', "w") #output file
if answer: #if a* finds path
    print(f"Path: {' -> '.join(answer[1])}")
    print(f"Total distance: {answer[0]} km")
    output.write(f"Path: {' -> '.join(answer[1])}\nTotal distance: {answer[0]} km")
else:
    print("No Path found")
    output.write("No Path found\n")

file_input.close()
output.close()
