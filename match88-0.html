from queue import PriorityQueue


def a_star_algo(graph,start,goal):
    nodesList = [(0,start)]
    parentCount = {}

    g_N = {}
    f_N = {}

    for node in graph:
        g_N[node] = float('inf')
        f_N[node] = float('inf')

    g_N[start] = 0
    f_N[start] = graph[start]['heuristic']

    while nodesList:
        curr = nodesList.pop(0)[1]
        if curr == goal:
            path = []
            while curr in parentCount:
                path.append(curr)
                curr = parentCount[curr]
            path.append(start)
            path.reverse()
            totalDist = g_N[goal]



            return f"path: {' -> '.join(path)}\nTotal distance: {totalDist} km"

        for neighbour, distance in graph[curr]['neighbours'].items():
            approx_g_N = g_N[curr] + distance

            if approx_g_N < g_N[neighbour]:


                parentCount[neighbour] = curr
                g_N[neighbour] = approx_g_N
                f_N[neighbour] = g_N[neighbour] + graph[neighbour]['heuristic']
                nodesList.append((f_N[neighbour], neighbour))


                nodesList.sort()

    return 'NO PATH FOUND'


def input_file(filename):
    data = {}
    with open(filename, 'r') as file:
        for i in file:
            portions = i.split()
            node = portions[0]
            heuristic = int(portions[1])
            neighbours = {}

            for i in range(2, len(portions), 2):


                neighbour = portions[i]
                dist = int(portions[i+1])
                neighbours[neighbour] = dist

            data[node] = {'heuristic': heuristic, 'neighbours': neighbours}

    return data


def main():
    data = input_file('/content/Input file.txt')
    startNode = input('Start node: ')
    endNode = input('Destination: ')

    finalResult = a_star_algo(data,startNode,endNode)
    print(finalResult)

    with open('output_file.text','w') as file:
      file.write(finalResult)
if __name__ == "__main__":
    main()
