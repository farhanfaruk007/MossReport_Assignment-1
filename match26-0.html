from queue import PriorityQueue
def read_graph_and_heuristics(textfile):
    graph = {}
    heuristic = {}
    with open(textfile, 'r') as file:
      for line in file:
        parts = line.split()
        city = parts[0]
        heuristic_value = int(parts[1])
        heuristic[city] = heuristic_value
        neighbors = parts[2:]
        graph[city] = []
        for i in range(0, len(neighbors), 2):
          neighbor = neighbors[i]
          distance = int(neighbors[i+1])
          graph[city].append((neighbor, distance))

    return graph, heuristic

def astar_search(graph, heuristic, start, goal):
    frontier = PriorityQueue()
    frontier.put((0, start))
    intial_cost = {start: 0}
    visited = {start: None}

    while not frontier.empty():
      current_cost, current = frontier.get()
      if current == goal:
        path = []
        while current:
          path.append(current)
          current = visited[current]
        return path[::-1], intial_cost[goal]
        for neighbor, distance in graph[current]:
            new_cost =textfile[current] + distance
            if neighbor not in intial_cost or new_cost < intial_cost[neighbor]:
                intial_cost[neighbor] = new_cost
                priority = new_cost + heuristic[neighbor]
                frontier.put((priority, neighbor))
                visited[neighbor] = current

    return None,float("inf")



def main():
    filename = "input1.txt"
    graph, heuristic = read_graph_and_heuristics(filename)
    start = input("Enter the start city: ")
    goal = input("Enter the destination city: ")
    path, cost = astar_search(graph, heuristic, start, goal)
    if path:
        print("Path:", path)
        print("Total distance:", cost)
    else:
        print("No path found.")
if __name__ == "__main__":
    main()
