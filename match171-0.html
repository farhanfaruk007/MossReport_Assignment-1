import heapq

def load_heuristic_values(filepath):
    heuristics = {}
    try:


        with open(filepath, 'r') as file:
            for line in file:
                parts = line.strip().split()
                if len(parts) >= 2:
                    heuristics[parts[0]] = int(parts[1])
    except FileNotFoundError:
        print(f"Error: The file '{filepath}' does not exist.")
    return heuristics


def load_graph(filepath):
    graph = {}
    try:
        with open(filepath, 'r') as file:
            for line in file:
                parts = line.strip().split()
                node = parts[0]
                edges = {}

                for i in range(2, len(parts), 2):
                    neighbor = parts[i]
                    distance = int(parts[i + 1])
                    edges[neighbor] = distance
                graph[node] = edges
    except FileNotFoundError:
        print(f"Error: The file '{filepath}' does not exist.")
    return graph


def A_star_search(start, goal, graph, heuristics, max_iterations=1000):
    priority_queue = []
    heapq.heappush(priority_queue, (0, start))
    path = {start: None}
    cost_to_node = {start: 0}

    for g in range(max_iterations):
        if not priority_queue:
            print("No more nodes to explore.")
            break

        g, current = heapq.heappop(priority_queue)



        if current == goal:
            break

        for neighbor, weight in graph.get(current, {}).items():
            new_cost = cost_to_node[current] + weight
            if neighbor not in cost_to_node or new_cost < cost_to_node[neighbor]:
                cost_to_node[neighbor] = new_cost
                priority = new_cost + heuristics.get(neighbor, float('inf'))
                heapq.heappush(priority_queue, (priority, neighbor))
                path[neighbor] = current
    else:
        print("Reached maximum iterations without finding a solution.")
        return None, None

    return path, cost_to_node


def display_path(path, cost_to_node, start, goal, max_iterations=100):
    route = []
    current = goal

    for _ in range(max_iterations):
        if current is None:
            break
        route.append(current)
        current = path.get(current)
    route.reverse()

    if not route or route[0] != start:
        print("No path could be found.")
    else:
        print("Path:", " --> ".join(route))
        print("Total Distance:", cost_to_node[goal], "km")


def main():
    file_path = "/content/Input file.txt"


    heuristics = load_heuristic_values(file_path)
    graph = load_graph(file_path)

    start = input("Enter the starting node: ").strip()
    goal = input("Enter the goal node: ").strip()

    if start not in heuristics or goal not in heuristics:
        print("Starting or goal node not found in heuristic data.")
        return

    path, cost_to_node = A_star_search(start, goal, graph, heuristics)
    if path and cost_to_node:
        display_path(path, cost_to_node, start, goal)


if __name__ == "__main__":
    main()
