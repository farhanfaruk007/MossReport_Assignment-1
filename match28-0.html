def a_st_sr(intl, goal) :
    exp_list = []
    heapq.heappush(exp_list, (0, intl))
    p_list = {}
    g_score = {city : float('inf') for city in romania_map}
    g_score[intl] = 0
    f_score = {city : float('inf') for city in romania_map}
    f_score[intl] = strt_ld[intl]
    while exp_list :
        current = heapq.heappop(exp_list)[1]
        if current == goal :
            path = recons_path(p_list, current)
            return path, g_score[goal]
        for nbr, cost in romania_map[current] :
            apprx_g_score = g_score[current] + cost
            if apprx_g_score < g_score[nbr] :
                p_list[nbr] = current
                g_score[nbr] = apprx_g_score
                f_score[nbr] = g_score[nbr] + strt_ld[nbr]
                heapq.heappush(exp_list, (f_score[nbr], nbr))
    return None, None


def recons_path(p_list, current) :
    path = [current]
    while current in p_list :
        current = p_list[current]
        path.append(current)
    path.reverse()
    return path
