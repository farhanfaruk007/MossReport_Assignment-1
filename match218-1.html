def Astar(start,goal,parent,heuristic):
    heap=[]
    heapq.heappush(heap,(0,start))

    startNode={}
    cost={}

    startNode[start]= None
    cost[start]= 0

    while heap:
        currentValue= heapq.heappop(heap)[1]

        if currentValue == goal:
            break

        for neighbour in parent[currentValue]:
            funG= cost[currentValue]+ parent[currentValue][neighbour]

            if neighbour not in cost or funG< cost[neighbour]:
                cost[neighbour]= funG



                funF= funG+heuristic[neighbour]
                heapq.heappush(heap,(funF,neighbour))

                startNode[neighbour]= currentValue
    return startNode, cost
