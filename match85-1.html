import heapq


read_file = open("/content/input_1.txt")
graph_create={}
heuristic_val = {}
list_m = read_file.readlines()

for i in list_m:
    node_list = i.strip().split()
    node_name = node_list[0]
    heur_value = int(node_list[1])
    heuristic_val[node_name] = heur_value
    graph_create[node_name] = {}

    for j in range(2, len(node_list), 2):
        neighbor = node_list[j]
        cost = int(node_list[j + 1])
        graph_create[node_name][neighbor] = cost



def A_star(start, goal, h_data, graph):
    p_que = []
    par_map = {}
    h_val = h_data
    predecessors = {}
    route = {}
    heapq.heappush(p_que, (0, start))
    predecessors[start] = None
    route[start] = 0
    found = 1
    while found == 1 and p_que:
        curr_node = heapq.heappop(p_que)[1]
        if curr_node == goal:
          found=0
        else:
          for neighbor in graph[curr_node]:
            g_val = route[curr_node] + graph[curr_node][neighbor]

            if neighbor not in route or g_val < route[neighbor] :
              route[neighbor] = g_val
              f_val = g_val + h_val[neighbor]
              heapq.heappush(p_que, (f_val, neighbor))
              predecessors[neighbor] = curr_node

    return predecessors,route




start_n = input("Please enter start destination: ")
goal = input("Please enter end destination: ")
predecessors, route = A_star(start_n, goal, heuristic_val, graph_create)
opt_way = []
opt_way.append(goal)
temp = goal

while temp != start_n:
    opt_way.append(predecessors[temp])
    temp = predecessors[temp]
n_val = len(opt_way)
if len(opt_way) != 0:


    res = "Path: "
    for i in range(n_val - 1, -1, -1):
        if opt_way[i] == goal:
            res += opt_way[i]
        else:
            res += opt_way[i] + " --> "
    print(res)
    print("Total distance: ", route[goal], "km")
else:
    print("No Path Found")

read_file.close()
