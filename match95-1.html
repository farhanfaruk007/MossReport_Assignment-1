from queue import PriorityQueue

def a_star(data, start, destination):
    open_set = PriorityQueue()
    open_set.put((data[start]['Heuristic'], start))
    explored = set()
    
    cost_to_node = {start: 0}
    estimated_cost = {start: data[start]['Heuristic']}
    path = {}

    while not open_set.empty():


        current = open_set.get()[1]
        if current == destination:
            route = [destination]
            while current in path:
                current = path[current]
                route.append(current)
            route.reverse()
            return estimated_cost[destination], route
        explored.add(current)



        for neighbor, distance in data[current]["Neighbor"].items():
            new_cost = cost_to_node[current] + distance
            if neighbor in explored:
                continue
            if neighbor not in cost_to_node or new_cost < cost_to_node[neighbor]:
                cost_to_node[neighbor] = new_cost
                estimated_cost[neighbor] = new_cost + data[neighbor]['Heuristic']
                open_set.put((estimated_cost[neighbor], neighbor))
                path[neighbor] = current
    return None


data = {}
with open('input_file.txt', 'r') as file:
    for line in file:
        parts = line.split()
        city, heuristic = parts[0], int(parts[1])
        neighbors = {parts[i]: float(parts[i+1]) for i in range(2, len(parts), 2)}
        data[city] = {'Heuristic': heuristic, 'Neighbor': neighbors}

start = input('Start Node: ')
destination = input('Destination Node: ')


result = a_star(data, start, destination)
if result:
    distance, route = result
    print(f"Path: {' -> '.join(route)}")
    print(f"Total Distance: {distance} km")
else:
    print("NO PATH FOUND")
