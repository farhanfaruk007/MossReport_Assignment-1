import heapq

# Queue for nodes
class PQ:
    def __init__(self):
        self.nodes = []
    def push(self, node, cost):
        heapq.heappush(self.nodes, (cost, node))
    def pop(self):
        return heapq.heappop(self.nodes)[1]
    def is_empty(self):
        return len(self.nodes) == 0

class Node:
    def __init__(self, node, distance):
        self.node = node
        self.distance = distance

graph = {}
h_values = {}

def parse_input(file_path):
    with open(file_path, 'r') as f:
        for line in f:
            parts = line.split()
            if len(parts) < 3:
                continue
            node = parts[0]
            h_values[node] = int(parts[1])
            graph[node] = []


            for i in range(2, len(parts), 2):
                neighbor = parts[i]
                distance = int(parts[i + 1])
                graph[node].append(Node(neighbor, distance))

                                       # A* search alg
def astar(start, end):
    path = {}
    dist = {}
    q = PQ()
    q.push(start, 0)
    dist[start] = 0
    path[start] = None
    found = False

    while not q.is_empty():
        current = q.pop()

        if current == end:
            found = True
            break


        for neighbor in graph.get(current, []):
            g = dist[current] + neighbor.distance
            if neighbor.node not in dist or g < dist[neighbor.node]:
                dist[neighbor.node] = g
                f = g + h_values.get(neighbor.node, float('inf'))
                q.push(neighbor.node, f)
                path[neighbor.node] = current
