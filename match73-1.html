import heapq

class Graph:
    def __init__(self):
        self.graph = {}

    def add_node(self, node, heuristic, neighbors):
        self.graph[node] = {"heuristic": heuristic, "neighbors": neighbors}



    def Astar_search(self, start, goal):
        open_list = []
        g_costs = {node: float('inf') for node in self.graph}
        g_costs[start] = 0
        prev_nodes = {}



        heapq.heappush(open_list, (self.graph[start]["heuristic"], start))

        while open_list:
            current_f, current = heapq.heappop(open_list)
            if current == goal:
                path = []
                while current in prev_nodes:
                    path.append(current)
                    current = prev_nodes[current]
                path.append(start)
                path.reverse()
                return path, g_costs[goal]

            for neighbor, distance in self.graph[current]["neighbors"]:
                tentative_g_cost = g_costs[current] + distance
                if tentative_g_cost < g_costs[neighbor]:
                    prev_nodes[neighbor] = current
                    g_costs[neighbor] = tentative_g_cost
                    f_cost = tentative_g_cost + self.graph[neighbor]["heuristic"]


                    heapq.heappush(open_list, (f_cost, neighbor))

        return "NO PATH FOUND", None

def graph_file(file_name):
    graph = Graph()
    with open(file_name, 'r') as file:
        for line in file:
            parts = line.split()
            node = parts[0]
            heuristic = int(parts[1])


            neighbors = [(parts[i], int(parts[i + 1])) for i in range(2, len(parts), 2)]
            graph.add_node(node, heuristic, neighbors)
    return graph

def main():
    file_name = "/content/drive/MyDrive/422_lab_1/input.txt"
    out_file = "/content/drive/MyDrive/422_lab_1/output.txt"
    graph = graph_file(file_name)

    start = input("Enter the start node: ")
    destination = input("Enter the destination node: ")

    path, total_distance = graph.Astar_search(start, destination)

    with open(out_file, 'w') as f:
        if path == "NO PATH FOUND":
            f.write(path)
        else:
            f.write("Path: " + " -> ".join(path) + "\n")
            f.write("Total distance: " + str(total_distance) + " km\n")

    print(f"Results written to {out_file}")
if __name__ == "__main__":
    main()
