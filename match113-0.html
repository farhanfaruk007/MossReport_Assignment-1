from queue import PriorityQueue
def astar_algo(graph,src,des):
    v_node=PriorityQueue()
    v_node.put((graph[src]['Heuristic'],src))
    e_node=set()
    w={src:0}
    est_d={src:graph[src]['Heuristic']}
    path={}
    while not v_node.empty():
        if src==des:
            return (w[src],src)
        c_node=v_node.get()[1]

        if c_node==des:

            backtrack=[des]
            while c_node in path:
                c_node=path[c_node]
                backtrack.append(c_node)
            backtrack.reverse()
            return (est_d[des],backtrack)
        e_node.add(c_node[1])
        for n,info in graph[c_node]["Neighbor"].items():

            path_cost=w[c_node]+info
            if n in e_node:
                continue
            elif n not in w or path_cost<w[n]:

                w[n]=path_cost
                est_d[n]=path_cost+graph[n]['Heuristic']

                v_node.put((est_d[n],n))
                path[n]=c_node
    return



graph={}
inp_file=open('Input file.txt','r')
lines=inp_file.readlines()
for line in lines:
    info=line.split()
    city, heuristic,  n_info= info[0],int(info[1]), {}
    n_i=2
    while n_i<len(info):
        n_city=info[n_i]
        dist=float(info[n_i+1])
        n_info[n_city]=dist
        n_i+=2
    graph[city]={'Heuristic':heuristic,'Neighbor':n_info}

s_node=input('Start Node: ')
dest=input('Destination: ')
result=astar_algo(graph,s_node,dest)
if result==None:
    print('NO PATH FOUND')
else:
    ans='Path: '
    for i in result[1]:
        ans+=str(i)+' -> '
    ans=ans[:-4]+f'\nTotal Distance: {result[0]} km'
    print(ans)
