from queue import PriorityQueue

def A_star_search(s_city, e_city, h_info, neigh_info):
    val = PriorityQueue()
    initial_cost=0+h_info[s_city]
    initial_distance=0
    initial_city=s_city
    initial_path=[s_city]
    val.put([initial_cost, initial_distance, initial_city,initial_path])
    while not val.empty():
        info=val.get()
        t_cost=info[0]
        n_distance=info[1]
        c_city=info[2]
        c_path_list=info[3]
        if c_city != e_city:
            neighbours_list=neigh_info[c_city]
            for neigh, neigh_dist in neighbours_list.items():
                new_path=update_path(neigh,neigh_dist,n_distance,c_path_list)
                val.put(new_path)
        elif c_city == e_city:
            return [n_distance,c_path_list]
        else:
            return [None, None]

def update_path(a,b,c,d):
  n_path_list = d + [a]
  n_neigh_dist = c + b
  n_t_cost =  h_info[a]+n_neigh_dist
  return [ n_t_cost ,n_neigh_dist,a,n_path_list]

city_info=open('Input_file.txt')
h_info={}
neigh_info={}
total_cities=25
for cities in range(total_cities):
  line = city_info.readline()
  if not line:  # Check if the line is empty
    break  # Exit the loop if there are no more lines
  lines = line.strip().split()
  h_info[lines[0]] = int(lines[1])
  neigh_info[lines[0]] = {}
  for neigh in range(2, len(lines), 2):
    neigh_info[lines[0]][lines[neigh]] = int(lines[neigh + 1])


start=input("Enter your starting city: ")
end=input("Enter your destination  city: ")
total_journey_distance,journey_map= A_star_search(start, end,h_info ,neigh_info)

if (total_journey_distance==None) and (journey_map==None):
  print(f"NO PATH FOUND")
else:
  path=''
  for cities in range(len(journey_map)):
    if cities<len(journey_map)-1:
      path+=journey_map[cities]+'-> '
    else:
      path+=journey_map[cities]
  print("Path: ",path)
  print("Total distance:",total_journey_distance,"KM")
a1=open("output.txt",'w')
a1.write('Path: '+path+'\n')
a1.write('Total distance: '+str(total_journey_distance))
a1.close()
