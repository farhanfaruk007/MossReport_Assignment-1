n=open(r"C:\Users\User\Downloads\Input file.txt",'r')
dict={}

lines = n.readlines()
n.close()


for line in lines:


    hold = line.strip().split()

    dict[hold[0]]=[int(hold[1]),[]]
    for j in range(2,len(hold),2):

      dict[hold[0]][1].append((hold[j],int(hold[j+1])))


start= input("Enter the starting point: ")
end= input("Enter the destination point: ")


from queue import PriorityQueue


q = PriorityQueue()

temp= dict[start]

q.put([temp[0],[start],0])


def pathsearch(array,cost, start, end):


  store=""
  for i in range(len(array)):
    if i!=len(array)-1:
      store+=array[i]+", "
    else:
      store+=array[i]
  return f'Optimal Path from {start} to {end} is {store} with cost of {cost} km'



visited=[]
path_found=False
while not q.empty():
  deque= q.get()

  node=deque[1][len(deque[1])-1]
  cost=deque[2]
  visited.append(node)

  if node==end:
    path_found=True

    print(pathsearch(deque[1],cost, start, end))
    break



  for tup in dict[node][1]:

    if tup[0] not in visited:

      pathcost=tup[1]+cost
      huristic=dict[tup[0]][0]
      totalcost=huristic+ pathcost

      copy=deque[1].copy()
      new=[totalcost,copy,pathcost]

      new[1].append(tup[0])
      q.put(new)
if path_found==False:
  print("NO PATH FOUND")
