import heapq

class Graph:
    def __init__(self):
        self.adjLst = {}

    def addEdges(self, nodeOne, nodeTwo, dis=0):
        if nodeOne not in self.adjLst:
            self.adjLst[nodeOne] = []
        
        if nodeTwo not in self.adjLst:
            self.adjLst[nodeTwo] = []

        if (nodeOne,dis) not in self.adjLst[nodeTwo]:
            self.adjLst[nodeTwo] += [(nodeOne, dis)]

        if (nodeTwo,dis) not in self.adjLst[nodeOne]:
            self.adjLst[nodeOne] += [(nodeTwo, dis)]

    def getNeighbors(self, node):
        return self.adjLst[node] if node in self.adjLst else "<!Node not Found!>"



    def getGraph(self):
        return self.adjLst

def heuristic(node, heuristicList) -> int:
    for n, hval in heuristicList:
        if n == node:
            return hval
        
    return 0   

def aStarSearch(graph, start, goal, heuristicList):
    notvstd = [(0, start)]   
    vstd = set()             
    cost = {start: 0}          
    par = {}                

    while notvstd:
        x, curr = heapq.heappop(notvstd)

        if curr == goal:
            path = []
            totalCost = cost[curr]
            while curr in par:
                path.append(curr)
                curr = par[curr]
            path.append(start)

            return list(reversed(path)), totalCost  

        vstd.add(curr)

        for nbr, nbrCost in graph.getNeighbors(curr):
            tentativeCost = cost[curr] + nbrCost

            if nbr in vstd or tentativeCost >= cost.get(nbr, float('inf')):
                continue

            if tentativeCost < cost.get(nbr, float('inf')):
                par[nbr] = curr
                cost[nbr] = tentativeCost
                totalCost = tentativeCost + heuristic(nbr, heuristicList)
                heapq.heappush(notvstd, (totalCost, nbr))

    return None, 0  



def getInput(filename):
    with open(filename, 'r') as f:
        return f.readlines()

if __name__ == '__main__':
    nodeInputs = getInput("input.txt")
    graph = Graph()
    heuristicList = []

    for nodeInfo in nodeInputs:
        nodeInfo = nodeInfo.split(' ')
        nodeList = [(nodeInfo[i], int(nodeInfo[i+1].rstrip())) for i in range(0, len(nodeInfo), 2)]
        node = nodeList[0][0]
        hval = nodeList[0][1]
        heuristicList.append((node, hval))


        neighbors = nodeList[1:]
        for nbr in neighbors:
            dis = nbr[1]
            graph.addEdges(node, nbr[0], dis)

    start = input("Start node: ")
    goal = input("Destination: ")
    path, totalCost = aStarSearch(graph, start, goal, heuristicList)

    
    if path:
        pathString = ' -> '.join(path)
        print(f"Path: {pathString}")
        print(f"Total disance: {totalCost} km")
