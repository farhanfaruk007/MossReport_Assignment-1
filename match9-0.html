inp = open("input.txt","r")

initial_table = {}
nei = {}

parent_li = []
parent_li_idx = []

info = inp.readlines()

path_flag = True


for i in range(len(info)) :
    line = info[i]
    item = line.split()
    initial_table[item[0]] = int(item[1])
    nei[item[0]] = []
    for j in range(2,len(item),2):
        nei[item[0]].append([item[j],int(item[j+1])])



start = input("Enter the star city: ",)


end = input("Enter the end city: ",)



A_li = []
a = start
b = initial_table[start]
A_li.append([b,a])
pop_node = None
# print(A_li)
while pop_node != end :

    if len(A_li) == 0 :
        path_flag = False
        break
    pop = A_li.pop(0)

    pop_node = pop[1]
    pop_val = pop[0]
    neighbour = nei[pop_node]

    for city in neighbour:
        city_name = city[0]   ##

        city_dis_own = city[1]
        dis = pop_val - initial_table[pop_node]

        city_dis = dis + city_dis_own + initial_table[city_name] ##

        if city_name not in parent_li_idx :

            A_li.append([city_dis,city_name])
            parent_li_idx.append(city_name)
            parent_li.append(pop_node)


        else :
            for i in range(len(A_li)):
                if city_name == A_li[i][1] :
                    xx =  A_li[i][0]
                    idx = i

                    if xx > city_dis :
                        A_li.pop(idx)
                        A_li.append([city_dis,city_name])
                        a = parent_li_idx.index(city_name)
                        parent_li_idx.pop(a)
                        parent_li.pop(a)
                        parent_li_idx.append(city_name)
                        parent_li.append(pop_node)


    A_li.sort()


if path_flag == False :
    print("NO PATH FOUND!!!")
else :
    path = []

    path.append(end)

    baap = None

    while baap != start :

        idx = parent_li_idx.index(end)
        baap = parent_li[idx]
        path.append(baap)
        end = baap

    path = path[::-1]
    # print(path)
    print("Path:", " -> ".join(path))
    print(pop_val,"kM")
