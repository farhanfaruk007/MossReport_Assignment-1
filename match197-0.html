import heapq

def graph_data(file_path):
    city_graph = {}
    h_values = {}

    with open(file_path, "r") as file:
        for line in file:
            parts = line.strip().split()


            node = parts[0]
            heuristic = int(parts[1])
            h_values[node] = heuristic
            neighbors = [(parts[vals], int(parts[vals + 1])) for vals in range(2, len(parts), 2)]
            city_graph[node] = neighbors

    return city_graph, h_values

def a_star_search(city_graph, h_values, start, goal):
    priority_queue = []
    heapq.heappush(priority_queue, (0 + h_values[start], 0, start, [start]))
    visited = set()

    while priority_queue:
        _, current_cost, current, path = heapq.heappop(priority_queue)

        if current in visited:
            continue

        if current == goal:
            return path, current_cost

        visited.add(current)

        for neighbor, cost in city_graph[current]:
            if neighbor not in visited:
                new_current_cost = current_cost + cost
                total_cost = new_current_cost + h_values[neighbor]
                heapq.heappush(priority_queue, (total_cost, new_current_cost, neighbor, path + [neighbor]))

    return None, None



def main():

    file_path = "city_info.txt"
    city_graph, h_values = graph_data(file_path)

    start = input("Start node: ")
    goal = input("Destination: ")

    if start not in city_graph or goal not in city_graph:
        print("Invalid start or destination node.")
        return

    path, total_distance = a_star_search(city_graph, h_values, start, goal)

    if path:
        print(f"Path: {' -> '.join(path)}")
        print(f"Total distance: {total_distance} km")
    else:
        print("NO PATH FOUND")

if __name__ == "__main__":
    main()
