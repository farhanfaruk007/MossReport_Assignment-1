input_file= open("Input file.txt","r")
output_file= open("output.txt","w")

graph = {}
heuristic_set= {}
start_node = input("Start node: ")
destination = input("Destination: ")

for line in input_file:
        node,h_value,*adjacentNodes= line.split() #remaining all nodes are adjacent nodes
        heuristic_set[node] = int(h_value)
        adjacents = []
        for i in range(0, len(adjacentNodes), 2):
            adj_node = adjacentNodes[i]
            adj_weight = int(adjacentNodes[i + 1])
            adjacents.append((adj_node, adj_weight))

        graph[node] = adjacents

import heapq



def a_star_search(graph, start, goal, heuristic_set):
    pr_queue = [(0, start)]
    parent_node = {start: (None, 0)}
    while pr_queue:
        curr_cost, curr_node = heapq.heappop(pr_queue)
        if curr_node == goal:
            path = []
            while curr_node is not None:
                path.append(curr_node)
                curr_node = parent_node[curr_node][0]
            path.reverse()
            return path, parent_node[goal][1]
        for (neighbor, cost) in graph.get(curr_node, []):
            new_cost = parent_node[curr_node][1] + cost
            if neighbor not in parent_node or new_cost < parent_node[neighbor][1]:
                parent_node[neighbor] = (curr_node, new_cost)
                heapq.heappush(pr_queue, (new_cost + heuristic_set.get(neighbor, 0), neighbor))
    return (None, None)


if (start_node not in graph) or (destination not in graph):


    output_file.write("Wrong intput")
else:
    path, total_distance = a_star_search(graph, start_node, destination, heuristic_set)
    if path is None:
        output_file.write("No path found!")
    else:
       output_file.write("Path found: " + " -> ".join(path) + "\n")
       output_file.write("Total distance: " + str(total_distance) + " km")
