import heapq
def create_graph(input_file):
    graph = {}
    heuristics = {}

    with open(input_file, 'r') as file:
        for line in file:
            parts = line.split()
            city = parts[0]


            heuristic = int(parts[1])
            neighbors = parts[2:]

            heuristics[city] = heuristic

            graph[city] = []

            for i in range(0, len(neighbors), 2):
                neighbor_city = neighbors[i]
                weight = int(neighbors[i + 1])


                graph[city].append((neighbor_city, weight))

    return graph, heuristics


def astar(h, gr, g, s):
    pq = []
    heapq.heappush(pq, (h[s], 0, s, [s]))


    visited = set()

    while pq:

        f_cost, g_cost, current, path = heapq.heappop(pq)

        if current == g:
            return path, g_cost


        if current in visited:
            continue
        visited.add(current)


        for neighbor, weight in gr.get(current, []):
            if neighbor not in visited:
                new_g_cost = g_cost + weight
                new_f_cost = new_g_cost + h[neighbor]
                heapq.heappush(pq, (new_f_cost, new_g_cost, neighbor, path + [neighbor]))


    return None, float('inf')


input_file = "input file.txt"
graph, heuristics = create_graph(input_file)
start = 'Arad'
goal = 'Bucharest'
path, cost = astar(heuristics, graph, goal, start)
if path is None:
    print('No path found')
else:
    print(f"Path: {path}")
    print(f"Cost: {cost}")
