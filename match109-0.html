import heapq
input_file = open("/content/sample_data/Input_file.txt", "r")

def heuristic_Value(input_file):
    heuristic_own = {}
    heuristic_neighbor = {}

    for each_line in input_file:
        cities = each_line.strip().split()
        city_name = cities[0]
        heuristic_own[city_name] = int(cities[1])

        neighbors = {}
        for i in range(2, len(cities), 2):
            neighbor_city = cities[i]
            distance = int(cities[i + 1])
            neighbors[neighbor_city] = distance


        heuristic_neighbor[city_name] = neighbors



    return heuristic_own, heuristic_neighbor

def a_star(start_city, end_city, heuristic_neighbor, heuristic_own):
    track_city = []
    heapq.heappush(track_city, (0, start_city))
    acl_path = {}
    cost = {}
    acl_path[start_city] = None
    cost[start_city] = 0


    while track_city:
        current_city = heapq.heappop(track_city)[1]

        if current_city == end_city:
            break

        for neighbor_city in heuristic_neighbor[current_city]:
            new_cost = cost[current_city] + heuristic_neighbor[current_city][neighbor_city]
            if neighbor_city not in cost or new_cost < cost[neighbor_city]:
                cost[neighbor_city] = new_cost
                priority = new_cost + heuristic_own[neighbor_city]


                heapq.heappush(track_city, (priority, neighbor_city))
                acl_path[neighbor_city] = current_city

    return acl_path, cost

def output(path, cost, start_city, end_city):
    current_city = end_city
    acl_path = []

    while current_city is not start_city:
      acl_path.append(current_city)
      new_city = path[current_city]
      current_city = new_city
    acl_path.append(start_city)

    if len(acl_path) != 0:
      acl_path.reverse()
      print(*acl_path, sep = " --> ")
    else:
      print("NO PATH FOUND!")
    print(f'Total distance: {cost[end_city]} km')

start_city = input("Enter the start node: ")
end_city = input("Enter the goal node: ")

if start_city and end_city not in heuristic_own:
    print("Start or goal does not exist.")
else:
    path, cost = a_star(start_city, end_city, heuristic_neighbor, heuristic_own)
    output(path, cost, start_city, end_city)
