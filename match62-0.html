def A_star(start, goal, graph, hn):

    path_list = []
    fn = 0 + hn[start]
    gn = 0
    path = [start]
    path_list = [(start, fn, gn, path)]
    visited = []

    while path_list:

        path_list.sort(key=lambda x: x[1])
        node, fn, gn, path = path_list.pop(0)

        if node not in visited:
          visited.append(node)

        if node == goal:
          return path, gn

        if node in graph:
          for adj_node, g_n in graph[node]:
            if adj_node not in visited:
                path_list.append((adj_node, gn + g_n + hn[adj_node], gn + g_n, path + [adj_node]))





input = open('/content/drive/MyDrive/422/Lab1/input.txt','r').readlines()
lst = []
hn = {}
graph = {}

for i in input:
  i = i.split()

  if i!=" ":
    node = i[0]
    h = int(i[1])
    hn[node] = h

  adj_node = i[2:]
  for i in range(0, len(adj_node), 2):
      distance = int(adj_node[i + 1])
      lst.append((node, adj_node[i], distance))

for root, child, gn in lst:
      if root not in graph:
        graph[root] = []
      graph[root].append((child, gn))
# print(graph)
# print(lst)

start = "Arad"
goal = "Bucharest"



path, cost = A_star(start, goal, graph, hn)

print("Path:", end=" ")

for i in range(len(path)):
    if i>0:
        print("->", end=" ")
    print(path[i], end=" ")

print('\n'"Total distance:", cost, "km")
