input_file = open("/content/drive/MyDrive/CSE422 LAB/CSE422_Lab_Assignment01_AStarSearch_Input.txt",mode = "r")

input_lines = input_file.readlines()

heus = {}
dic = {}

for i in input_lines:
  inp = i.split(" ")
  n = inp[0].strip()
  v = int(inp[1].strip())
  heus[n] = v
  vals = inp[2:]
  city = []
  cost = []
  for j in range(len(vals)):
    if j%2 == 0:
      city.append(vals[j].strip())
    else:
      cost.append(int(vals[j].strip()))
  lst = []
  idx = 0
  while idx < len(city):
    value = (city[idx],cost[idx])
    lst.append(value)
    idx+=1

  dic[n] = lst


class Graph:
    def __init__(self, adjacency_list):
        self.adjacency_list = adjacency_list
        self.dist = None

    def get_nodes(self, l):
        return self.adjacency_list[l]

    def heuristics(self,heuristics):
      self.heuristics = heuristics

    def heuristic(self, n):
        H = self.heuristics
        return H[n]



    def A_Star_Search(self, start, finish):
        opened = set([start])
        closed = set([])
        g = {}
        g[start] = 0
        parents = {}
        parents[start] = start

        while len(opened) > 0:
            n = None

            for l in opened:
                if n == None or g[l] + self.heuristic(l) < g[n] + self.heuristic(n):


                    n = l
            if n == None:
                print('NO PATH FOUND')
                return None

            if n == finish:
                reconstruct_path = []

                while parents[n] != n:
                    reconstruct_path.append(n)
                    n = parents[n]
                reconstruct_path.append(start)



                self.dist = g[finish]

                reconstruct_path.reverse()
                return reconstruct_path

            for (m, edge_weight) in self.get_nodes(n):

                if m not in opened and m not in closed:
                    opened.add(m)
                    parents[m] = n
                    g[m] = g[n] + edge_weight

                else:
                    if g[m] > g[n] + edge_weight:


                        g[m] = g[n] + edge_weight
                        parents[m] = n

                        if m in closed:
                          closed.remove(m)
                          opened.add(m)


            opened.remove(n)
            closed.add(n)

        print('NO PATH FOUND')

        return None

graph = Graph(dic)
graph.heuristics(heus)

s = 'Arad'
f = 'Bucharest'
result = graph.A_Star_Search(s, f)
print('Path:','-> '.join(result))

print("Total Distance :", graph.dist, 'km')
