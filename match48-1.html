import heapq

input_file = open("/content/input.txt", 'r')
graph = {}
hx = {}

# Reading input file and constructing graph and heuristics
for i in range(20):


    x = input_file.readline().split()
    hx[x[0]] = int(x[1])

    for j in range(2, len(x) - 1, 2):
        if x[0] not in graph:
            graph[x[0]] = {x[j]: int(x[j + 1])}
        else:
            graph[x[0]][x[j]] = int(x[j + 1])

input_file.close()



def a_star_search(start, goal, graph, hx):
    queue = []
    heapq.heappush(queue, (hx[start], 0, start))
    parent = {}
    path_cost = {node: float('inf') for node in graph}
    path_cost[start] = 0

    while queue:
        _, cost, node = heapq.heappop(queue)

        if node == goal:
            path = []
            while node in parent:
                path.append(node)
                node = parent[node]
            path.append(start)
            path.reverse()
            return path, path_cost[goal]


        if node in graph:
            for neighbor, edge_cost in graph[node].items():
                tentative_g_score = cost + edge_cost
                if tentative_g_score < path_cost[neighbor]:
                    path_cost[neighbor] = tentative_g_score
                    parent[neighbor] = node
                    f_score = tentative_g_score + hx[neighbor]
                    heapq.heappush(queue, (f_score, tentative_g_score, neighbor))


    return None, float('inf')


path, cost = a_star_search('Arad', 'Bucharest', graph, hx)
print("Path:", path)
print("Total Cost:", cost)
