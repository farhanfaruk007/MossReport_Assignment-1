import heapq


def a_star_search(start, target, graph_info, heuristic_valss):
    prior_queue = []
    heapq.heappush(prior_queue, (0, start))
    previous_node = {}
    update_cst = {}
    previous_node[start] = None
    update_cst[start] = 0

    while prior_queue:
        ongoing = heapq.heappop(prior_queue)[1]

        if ongoing == target:
            break

        for neighbor in graph_info[ongoing]:
            new_cst = update_cst[ongoing] + graph_info[ongoing][neighbor]
            if neighbor not in update_cst or new_cst < update_cst[neighbor]:
                update_cst[neighbor] = new_cst


                total_cost = new_cst + heuristic_valss[neighbor]
                heapq.heappush(prior_queue, (total_cost, neighbor))
                previous_node[neighbor] = ongoing

    return previous_node, update_cst
