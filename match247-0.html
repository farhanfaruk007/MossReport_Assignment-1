import heapq
input_file = open("/content/Input file.txt", "r").read()
heuristic_values = {}
adjacency_list = {}
lines = input_file.strip().split("\n")



for line in lines:
    elements = line.split(" ")
    heuristic_values[elements[0]] = int(elements[1])
    adjacency_list[elements[0]] = []
    for index in range(2, len(elements) - 1, 2):
        neighbor = elements[index]
        edge_weight = int(elements[index + 1])
        adjacency_list[elements[0]].append((neighbor, edge_weight))

start_node = input("Enter the Start node: ").strip()
goal_node = input("Enter the End node: ").strip()

def save_path(predecessors, source, destination, total_costs):
    # Check if a path exists
    if destination not in predecessors:
        with open("/content/Output file.txt", 'w') as file:
            file.write("NO PATH FOUND\n")
        return

    route = []
    current = destination
    while current is not None:
        route.append(current)
        current = predecessors[current]
    route.reverse()




    with open("/content/Output file.txt", 'w') as file:
        file.write(f"Path: {' -> '.join(route)}\n")
        file.write(f"Total distance: {total_costs[destination]} km\n")

def perform_astar(source, destination):
    # Initialize dictionaries for tracking
    predecessors = {source: None}
    total_costs = {source: 0}
    priority_queue = []
    heapq.heappush(priority_queue, (heuristic_values[source], source))

    while priority_queue:
        current_cost, current_node = heapq.heappop(priority_queue)
        if current_node == destination:
            break
        for neighbor, weight in adjacency_list[current_node]:
            new_cost = total_costs[current_node] + weight
            if neighbor not in total_costs or new_cost < total_costs[neighbor]:
                total_costs[neighbor] = new_cost
                estimated_cost = new_cost + heuristic_values[neighbor]
                heapq.heappush(priority_queue, (estimated_cost, neighbor))
                predecessors[neighbor] = current_node


    save_path(predecessors, source, destination, total_costs)


perform_astar(start_node, goal_node)
