import heapq

def a_star_search(graph, start, goal):

    priority_queue = [(graph[start]['heuristic'], start)]
    visited = set()
    predecessors = {}


    actual_cost = {}
    for node in graph:
        actual_cost[node] = float('inf')
    actual_cost[start] = 0

    while priority_queue:
        current = heapq.heappop(priority_queue)
        current_node = current[1]



        if current_node == goal:
            path = []
            while current_node in predecessors:
                path.insert(0, current_node)
                current_node = predecessors[current_node]
            path.insert(0, start)
            return path

        visited.add(current_node)

        neighbors = graph[current_node]['neighbors']
        for neighbor in neighbors:
            distance = neighbors[neighbor]
            new_cost = actual_cost[current_node] + distance



            if new_cost < actual_cost[neighbor]:
                actual_cost[neighbor] = new_cost
                predecessors[neighbor] = current_node
                estimated_total_cost = new_cost + graph[neighbor]['heuristic']
                heapq.heappush(priority_queue, (estimated_total_cost, neighbor))

    return None

graph = {}
with open("/content/sample_data/input.txt", 'r') as file:
    for line in file:
        data = line.split()
        node = data[0]
        heuristic = int(data[1])
        neighbors = {}
        for i in range(2, len(data), 2):


            neighbor = data[i]
            distance = int(data[i + 1])
            neighbors[neighbor] = distance
        graph[node] = {'heuristic': heuristic, 'neighbors': neighbors}


with open("/content/sample_data/Output.txt", 'w') as output_file:

    start_node = input("Enter your start node: ")
    destination_node = input("Enter your destination node: ")

    if start_node not in graph:
        print("Start node not found in the graph.", file=output_file)
    elif destination_node not in graph:
        print("Destination node not found in the graph.", file=output_file)
    else:
        path = a_star_search(graph, start_node, destination_node)

        if path:
            total_distance = sum(graph[path[i]]['neighbors'][path[i + 1]] for i in range(len(path) - 1))
            print("Path:", " -> ".join(path), file=output_file)
            print(f"Total distance: {total_distance} km", file=output_file)
        else:
            print("NO PATH FOUND", file=output_file)
