import heapq

#Part1:
input_file=open("/LAB 1 TXT.txt",'r')
q1 = input_file.readlines()
rom = {}
heuristic = {}


for path in q1:
  a = path.split()
  city = a[0]
  n = len(a)
  heuristic[city] = int(a[1])
  for i in range(2,n,2):
      if city not in rom:
          rom[city] = {}
      else:
          rom[city][a[i]]=int(a[i+1])

#Part2: Constructing the A* Function
def create_path(parent, end):
  path = []


  while end in parent:
      path.insert(0, end)
      end = parent[end]
  path.insert(0, end)
  return path

def A_star_Search(rom, heuristic, start, end):
  explored = []
  not_explored = []


  g = {start:0}
  parent = {}
  f_cost = heuristic[start] + g[start]
  heapq.heappush(not_explored, (f_cost,start))
  while not_explored:
    m = heapq.heappop(not_explored)
    if  m[1] == end:
      dist = g[end]
      return create_path(parent, end), dist
    explored.append(m[1])
    for neighbor, distance in rom[m[1]].items():
      if neighbor in explored:
          continue
      g_cost = distance + g[m[1]]
      if  neighbor in not_explored and g_cost<g[neighbor]:
        not_explored.remove(neighbor)
      if  neighbor in explored and g_cost<g[neighbor]:
        explored.remove(neighbor)
      if neighbor not in not_explored or neighbor not in explored:
        g[neighbor] = g_cost
        f_cost = g_cost + heuristic[neighbor]
        heapq.heappush(not_explored, (f_cost, neighbor))
        parent[neighbor] = m[1]
  return None, -1

#User Prompt/Function call/Printing Output:

start = input("Please enter the starting point: ")
end = input("Please enter your destination: ")
function_call, dist = A_star_Search(rom, heuristic, start, end)

if function_call == None:
  print("No path found")
else:
  print("Path: ", end = "")
  for i in range(len(function_call)):
      if i==len(function_call)-1:
          print(function_call[i])
      else:
          print(function_call[i], "--> ", end = "")
  print("Total Distance:", dist, "Km") # Output printing
