import heapq

with open("Input.txt", "r") as input_file:

    def heuristicValue(file):
        heuristic_own = {}
        heuristic_neighbor = {}

        for line in file:
            line = line.replace("\n", "")
            while "  " in line:
                line = line.replace("  ", " ")

            cities = []
            current_word = ""
            for char in line:
                if char == " ":
                    cities.append(current_word)
                    current_word = ""
                else:
                    current_word += char
            if current_word:
                cities.append(current_word)

            city = cities[0]
            heuristic_own[city] = int(cities[1])

            neighbors = {}
            for i in range(2, len(cities), 2):
                neighbor_city = cities[i]


                distance = int(cities[i + 1])
                neighbors[neighbor_city] = distance
            heuristic_neighbor[city] = neighbors

        return heuristic_own, heuristic_neighbor

    heuristic_own, heuristic_neighbor = heuristicValue(input_file)



def aStar(start, goal, neighbor_heuristics, own_heuristics):
    visited = []
    heapq.heappush(visited, (0, start))
    paths = {start: None}
    costs = {start: 0}

    while visited:
        current = heapq.heappop(visited)[1]

        if current == goal:
            break

        for neighbor, distance in neighbor_heuristics.get(current, {}).items():
            new_cost = costs[current] + distance
            if neighbor not in costs or new_cost < costs[neighbor]:
                costs[neighbor] = new_cost
                priority = new_cost + own_heuristics[neighbor]


                heapq.heappush(visited, (priority, neighbor))
                paths[neighbor] = current

    return paths, costs

def outputBuilder(path_dict, cost_dict, start, goal):
    path = []
    current = goal

    while current != start:
        path.append(current)
        current = path_dict[current]
    path.append(start)
    final_path = ' --> '.join(reversed(path))

    print("Path:", final_path)
    print("Total distance:", cost_dict[goal], "km")

start_city = input("Enter the start node: ")
goal_city = input("Enter the goal node: ")

if start_city in heuristic_own and goal_city in heuristic_own:
    path_data, cost_data = aStar(start_city, goal_city, heuristic_neighbor, heuristic_own)
    outputBuilder(path_data, cost_data, start_city, goal_city)
else:
    print("Start or goal does not exist.")
