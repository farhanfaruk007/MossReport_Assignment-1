from queue import PriorityQueue

def A_Star_Search(GraphDict, Main_Node, Destination_Node):
    PQ = PriorityQueue()
    PQ.put((0,Main_Node))
    Travel_Graph = {}

    Start_Point = {node: float('inf') for node in GraphDict}


    Start_Point[Main_Node] = 0

    Travel_cost = {node: float('inf') for node in GraphDict}
    Travel_cost[Main_Node] = GraphDict[Main_Node]['h']

    while not PQ.empty():
        Current_Priority, Current_Point = PQ.get()

        if (Current_Point == Destination_Node):


            Optimal_Path = []

            while Current_Point in Travel_Graph:
                Optimal_Path.append(Current_Point)
                Current_Point = Travel_Graph[Current_Point]

            Optimal_Path.append(Main_Node)
            Optimal_Path.reverse()
            return f"Path: {' -> '.join(Optimal_Path)}\nTotal distance: {Start_Point[Destination_Node]} km"


        for Neig, Dist in GraphDict[Current_Point]['NeighbourDict'].items():
            new_path = Start_Point[Current_Point] + Dist

            if new_path < Start_Point[Neig]:
                Travel_Graph[Neig] = Current_Point
                Start_Point[Neig] = new_path

                Travel_cost[Neig] = new_path + GraphDict[Neig]['h']
                PQ.put((Travel_cost[Neig], Neig))

    return "NO PATH FOUND"

def text_to_graph(File):
    GraphDict = {}
    for line in File:
        List = line.split()
        Main_Node = List[0]
        h = int(List[1])
        NeighbourDict = {}

        for i in range(2,len(List),2):


            NeighbourNode = List[i] 
            Distance = int(List[i+1])
            NeighbourDict[NeighbourNode] = Distance

        GraphDict[Main_Node] = {'h':h,'NeighbourDict':NeighbourDict}
    
    return GraphDict


Input_File = open('Input file.txt','r')
