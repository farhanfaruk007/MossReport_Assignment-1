class Graph:

    def __init__(self, parent_child_info_dict, heuristic_dict):
        self.parent_child_info_dict = parent_child_info_dict
        self.heuristic_dict = heuristic_dict
        self.path = ""
        self.path_cost = 0
        self.checker = False

    def childs(self, parent):
        child_info_list = self.parent_child_info_dict[parent]
        return child_info_list



    def A_star(self, start, destination):
        open = set([start])
        closed = set([])
        dist = {}
        dist[start] = 0
        parents = {}
        parents[start] = start

        while len(open) > 0:
            w = None
            for x in open:
                if w == None or dist[x] + self.heuristic_dict[x] < dist[w] + self.heuristic_dict[w]:


                    w = x;
            if w == None:
                print('Path does not exist!')
                return None
            if w == destination:
                path = []
                while parents[w] != w:
                    path.append(w)
                    w = parents[w]
                path.append(start)
                path.reverse()
                b = ""
                for i in range(len(path)):

                     if i != len(path)-1:
                        b += path[i]+"->"
                     else:
                        b += path[i]

                goal = path[len(path)-1]
                self.path = b


                self.path_cost = dist[goal]
                self.checker = True
                return path


            for (j, cost) in self.childs(w):

                if j not in open and j not in closed:
                    open.add(j)
                    parents[j] = w
                    dist[j] = dist[w] + cost
                else:
                    if dist[j] > dist[w] + cost:


                        dist[j] = dist[w] + cost
                        parents[j] = w
                        if j in closed:
                            closed.remove(j)
                            open.add(j)

            open.remove(w)
            closed.add(w)

        self.path='path does not exist'
        l = "path"
        return None
