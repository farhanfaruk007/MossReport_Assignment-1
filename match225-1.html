import heapq

class GRAPH:
    def __init__(self):
        self.adjacency_list = {}

    def addEdges(self, from_node, to_node, dist=0):
        if from_node not in self.adjacency_list:
            self.adjacency_list[from_node] = []

        if to_node not in self.adjacency_list:


            self.adjacency_list[to_node] = []

        if (from_node,dist) not in self.adjacency_list[to_node]:
            self.adjacency_list[to_node] += [(from_node, dist)]

        if (to_node,dist) not in self.adjacency_list[from_node]:
            self.adjacency_list[from_node] += [(to_node, dist)]

    def get_neighbors(self, node):
        # return the neighbors if the node is present in the graph
        return self.adjacency_list[node] if node in self.adjacency_list else "<!Node not Found!>"

    def get_graph(self):
        return self.adjacency_list

def heuristic(node, heuristic_list) -> int:
    # Retrives the Heuristic Values from heuristic_list for any node
    for n, hval in heuristic_list:
        if n == node:
            return hval

    return 0    #Default value

def a_star_search(graph, start, goal, heuristic_list):
    not_visited= [(0, start)]   # Priority queue of nodes to explore
    visited = set()             # Set of visited nodes
    cost = {start: 0}           # Cost from start node to current node
    parent = {}                 # Parent nodes for path reconstruction

    while not_visited:
        current = heapq.heappop(not_visited)[1]
        if current == goal:
            # Path found, reconstruction and return it with distance
            path = []
            total_dis = cost[current]
            while current in parent:
                path.append(current)
                current = parent[current]
            path.append(start)

            return list(reversed(path)), total_dis  #Path and total distance

        visited.add(current)

        for neighbor, neighbor_cost in graph.get_neighbors(current):
            tentative_cost = cost[current] + neighbor_cost

            if neighbor in visited or tentative_cost >= cost.get(neighbor, float('inf')):
                continue

            if tentative_cost < cost.get(neighbor, float('inf')):
                parent[neighbor] = current
                cost[neighbor] = tentative_cost
                totalCost = tentative_cost + heuristic(neighbor, heuristic_list)
                heapq.heappush(not_visited, (totalCost, neighbor))

    return None, 0  #No path found

def get_input(filename):
    with open(filename, 'r') as f:
        return f.readlines()

if __name__ == '__main__':
    node_inputs = get_input("input_file.txt")
    graph = GRAPH()
    heuristic_list = []

    for nodeInfo in node_inputs:
        nodeInfo = nodeInfo.split(' ')
        nodeList = [(nodeInfo[i], int(nodeInfo[i+1].rstrip())) for i in range(0, len(nodeInfo)-1, 2)]

        node = nodeList[0][0]
        hval = nodeList[0][1]
        heuristic_list.append((node, hval))
        neighbors = nodeList[1:]

        for neighbor in neighbors:
            dist = neighbor[1]
            graph.addEdges(node, neighbor[0], dist)

    start_city = "Arad"
    goal_city = "Bucharest"

    path, total_dis = a_star_search(graph, start_city, goal_city, heuristic_list)

    with open("output.txt", "w") as output_file:
        if path:
            output_file.write(f"Path: {' -> '.join(path)}\n")
            output_file.write(f"Total distance: {total_dis} km\n")
        else:
            output_file.write("NO PATH FOUND\n")
