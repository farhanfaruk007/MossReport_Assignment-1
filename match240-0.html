import heapq

def read_graph(file_path):
    graph = {}
    heuristics = {}
    
    with open(file_path, 'r') as file:
        for line in file:
            parts = line.strip().split()
            city = parts[0].strip()  
            heuristic = int(parts[1])
            heuristics[city] = heuristic
            neighbors = [(parts[i].strip(), int(parts[i + 1])) for i in range(2, len(parts), 2)]
            graph[city] = neighbors


    print(heuristics)
    print(graph)


    return graph, heuristics


def a_star_search(graph, heuristics, start, destination):
    open_list = []
    heapq.heappush(open_list, (0 + heuristics[start], 0, start, [start])) 
    visited = set()
    
    while open_list:
        Est_cost, actual_path_cost, current, path = heapq.heappop(open_list)
        
        if current in visited:
            continue
        visited.add(current)
        
        if current == destination:
            return path, actual_path_cost
        
        for neighbor, distance in graph.get(current, []):
            if neighbor not in visited:
                total_g = actual_path_cost + distance


                Est_cost = total_g + heuristics[neighbor]
                heapq.heappush(open_list, (Est_cost, total_g, neighbor, path + [neighbor]))
    
    return None, None


def main():
    file_path = 'input_data.txt' 
    graph, heuristics = read_graph(file_path)
    


    start = input("start node: ").strip()  
    destination = input("destination: ").strip()  
    

    path, total_distance = a_star_search(graph, heuristics, start, destination)
    
    if path:
        print("Path:", " -> ".join(path))
        print("Total distance:", total_distance, "km")
    else:
        print("NO PATH FOUND")

main()
