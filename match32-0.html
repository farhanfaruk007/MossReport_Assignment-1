import heapq
inp = open(("input.txt"),'r')
out = open("output.txt", 'w')



def A_star(graph,strt,dest):

    priority_queue=[(graph[strt]['heuristic'], strt)]
    visited_nodes=set()  #automatically sorting by this built in function (set())
    predecessors={}
    path_cost= {node: float('inf') for node in graph}
    path_cost[strt]=0

    while priority_queue:
        _, current_node = heapq.heappop(priority_queue)    #Tupple unpacking
        if current_node==dest:
            path=[]
            while current_node in predecessors:
                path.insert(0, current_node)
                current_node = predecessors[current_node]
            path.insert(0, strt)
            return path
        else:


            visited_nodes.add(current_node)

        for neighbor, distance in graph[current_node]['neighbors'].items():
            new_path_cost = path_cost[current_node] + distance
            if new_path_cost < path_cost[neighbor]:
                predecessors[neighbor] = current_node
                path_cost[neighbor] = new_path_cost
                total_cost_estimated = new_path_cost + graph[neighbor]['heuristic']
                heapq.heappush(priority_queue, (total_cost_estimated, neighbor))
    return None



graph = {}

for i in inp:
    x = i.split()
    node = x[0]
    h_value = int(x[1])    #the heuristic value
    neighbors = {}
    for i in range(2, len(x), 2):


        neighbor = x[i]
        distance = int(x[i + 1])
        neighbors[neighbor] = distance
    graph[node]={'heuristic':h_value, 'neighbors':neighbors}

#input values and get output accordingly:
start_node=input("Start node: ")


dest_node=input("Destination: ")

goal_test = A_star(graph, start_node, dest_node)
if goal_test:
    total_distance = sum(graph[node]['neighbors'][goal_test[i + 1]] for i, node in enumerate(goal_test[:-1]))
    print("Path:"," -> ".join(goal_test), file=out)
    print("Total distance:", total_distance, "km", file=out)
else:
    print("NO PATH FOUND", file=out)

out.close()
