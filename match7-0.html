output_dict = {}

  straight_line_distance[key]=values

  parent_city=after_split[0]

  output_dict[parent_city] = child_city

def astarsearch(start_city,destination_city):
  global output_dict, straight_line_distance
  fringe=[]
  list_city=[0, start_city]
  fringe.append(list_city)
  #initially the path cost of the start node is 0


  path_cost={start_city: 0}
  heuristic_cost = {start_city:straight_line_distance [start_city]}
  track_parent={}
  #print(fringe)
  #print(path_cost)
  #print(heuristic_cost)
  while fringe:
    sorted_fringe = sorted(fringe)
    node = sorted_fringe[0][1]
    #print(output_dict[node])
    fringe.pop(0)  # Remove the first element from the priority queue
    #print(fringe)
    for city, cost in output_dict[node]:
            #print("child",city)
            #print("cost",cost)
            child_node_finder(city, cost, node, path_cost, heuristic_cost,track_parent,fringe)
    optimal_path=[]
    if node ==destination_city:
      optimal_path.append(destination_city)
      while node in track_parent:
          optimal_path.append(track_parent[node])
          node = track_parent[node]
      return optimal_path


def child_node_finder(child, cost, current_node, path_cost, heuristic_cost,track_parent, fringe):

    total_cost = path_cost[current_node] + cost
    if child  in path_cost:
      pass

    if (child not in path_cost) or (total_cost < path_cost[child]):

        track_parent[child] = current_node
        path_cost[child] = total_cost
        distance_cost=int(straight_line_distance[child])
        heuristic_cost[child] = path_cost[child] + distance_cost
        fringe.append([heuristic_cost[child], child])
