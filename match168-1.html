import heapq


class PriorityQueue:
    def __init__(self):
        self.elements = []
    def push(self, city, cost):
        heapq.heappush(self.elements, (cost, city))
    def pop(self):
        return heapq.heappop(self.elements)[1]

    def is_empty(self):
        return not self.elements

class Node:
    def __init__(self, city, distance):
        self.city = city
        self.distance = distance

n = {}
heuristics = {}

def graph():
    with open("input.txt", 'r') as file:
        for line in file:
            parts = line.strip().split()
            main_city = parts[0]
            heuristic_cost = int(parts[1])
            heuristics[main_city] = heuristic_cost  # store heuristic cost
            n.setdefault(main_city, [])

            # Process neighbors in pairs of (city, distance)
            neighbors = parts[2:]
            for i in range(0, len(neighbors), 2):
                neighbor_city = neighbors[i]
                distance = int(neighbors[i + 1])
                n[main_city].append(Node(neighbor_city, distance))
                n.setdefault(neighbor_city, []).append(Node(main_city, distance))

def astar(start, goal):
    graph()
    queue = PriorityQueue()
    queue.push(start, 0)
    path, costs = {start: None}, {start: 0}
    expanded = []



    while not queue.is_empty():
        current = queue.pop()
        expanded.append(current)

        if current == goal:
            break

        for neighbor in n[current]:
            new_cost = costs[current] + neighbor.distance


            if neighbor.city not in costs or new_cost < costs[neighbor.city]:
                costs[neighbor.city] = new_cost
                queue.push(neighbor.city, new_cost + heuristics.get(neighbor.city, 0))


                path[neighbor.city] = current

    display_results(start, goal, path, costs, expanded)

def display_results(start, goal, path, costs, expanded):
    final_path = []
    city = goal

    while city:
        final_path.append(city)
        city = path.get(city)

    final_path.reverse()
    print("A* Algorithm")
    print(f"path: {' -> '.join(final_path)}")
    print(f"Total Cost: {costs[goal]}")

if __name__ == "__main__":
    astar("Arad", "Bucharest")
