#A star algorithm
import heapq

def astar(begin, reach_goal, ad_graph, heuristic):
    front = [] #Priority Queue
    heapq.heappush(front, (0, begin))
    parent = {}
    act_cost = {}
    parent[begin] = None
    act_cost[begin] = 0
    

    while front:
        running = heapq.heappop(front)[1]
        #If goal find at the begining then stop the process
        if running == reach_goal:
            break

        for i in ad_graph[running]:
            n_cost = act_cost[running] + ad_graph[running][i] #this is my g(n) calculation
            if i not in act_cost or n_cost < act_cost[i]:
                act_cost[i] = n_cost
                priority =  heuristic[i] +n_cost  #this is my f(n) calculation. f(n) = h(n)+g(n)


                heapq.heappush(front, (priority, i))
                parent[i] = running
                
    return parent, act_cost
