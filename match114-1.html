import heapq

def A_star_search(graph, start, goal):
  nodes = [(0, start)]
  pt = {}
  gn = {nodes: float('inf') for nodes in graph}
  gn[start] = 0
  fn = {nodes: float('inf') for nodes in graph}
  fn[start] = graph[start]['heuristic']
  while nodes:
    cost, i = heapq.heappop(nodes)
    if i == goal:
      path = []
      while i in pt:
        path.append(i)
        i = pt[i]
        path.append(start)
        path.reverse()


        return f"Path: {' -> '.join(path)}\nTotal Distance: {gn[goal]} km"

    for j, distance in graph[i]["neighbors"].items():
      temp_g = gn[i] + distance
      if temp_g < gn[j]:
        pt[j] = i
        gn[j] = temp_g
        fn[j] = temp_g + graph[j]['heuristic']


        heapq.heappush(nodes, (fn[j], j))
  return "No Path Found"

def read_graph(filename):
  graph = {}
  with open(filename, 'r') as file:
    for line in file:
      parts = line.split()
      node = parts[0]
      heuristic = int(parts[1])
      neighbors = {}
      for k in range(2, len(parts), 2):


        j = parts[k]
        distance = int(parts[k + 1])
        neighbors[j] = distance
        graph[node] = {'heuristic': heuristic, 'neighbors': neighbors}
  return graph
