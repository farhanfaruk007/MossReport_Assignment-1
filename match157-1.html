import heapq

def load_graph(city_path):
    graph = {}
    heuristic = {}
    with open(city_path, 'r') as file:
        for line in file:


            parts = line.split()
            node = parts[0]
            heuristic[node] = int(parts[1])
            graph[node] = [(parts[i], int(parts[i+1])) for i in range(2, len(parts), 2)]
    return graph, heuristic

def a_star_search(graph, heuristic, start, goal):
    open_list = []
    heapq.heappush(open_list, (0 + heuristic[start], start, [start], 0))
    visited = set()

    while open_list:
        _, current_city, path, actual_cost = heapq.heappop(open_list)

        if current_city in visited:
            continue
        visited.add(current_city)

        if current_city == goal:
            return path, actual_cost



        for neighbor, distance in graph.get(current_city, []):
            if neighbor not in visited:
                total_cost = actual_cost + distance + heuristic[neighbor]
                heapq.heappush(open_list, (total_cost, neighbor, path + [neighbor], actual_cost + distance))

    return "NO PATH FOUND", 0

def main():
    graph, heuristic = load_graph('input.txt')
    start = input("Start node: ")
    goal = input("Destination: ")
    path, total_distance = a_star_search(graph, heuristic, start, goal)
    if path == "NO PATH FOUND":
        print(path)
    else:
        print("Path:", " -> ".join(path))
        print("Total distance:", total_distance, "km")

if __name__ == "__main__":
    main()
