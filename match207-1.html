import heapq


with open("/content/drive/MyDrive/input.txt") as input_file:
    adj_list = {}
    heuristics = {}

    for line in input_file:
        elements = line.split()
        parent_node = elements[0]
        heuristic_value = int(elements[1])
        children = elements[2:]

        heuristics[parent_node] = heuristic_value
        neighbors = [(children[i], int(children[i + 1])) for i in range(0, len(children), 2)]
        adj_list[parent_node] = neighbors

def a_star_search(start, goal):
    # Priority queue for frontier
    frontier = []
    heapq.heappush(frontier, (heuristics[start], start))

    # Cost dictionary from start to each node


    actual_cost = {start: 0}
    path_tracker = {}

    while frontier:
        #lowest f(n) value node
        f_n, current_node = heapq.heappop(frontier)


        if current_node == goal:
            path = []
            while current_node in path_tracker:
                path.append(current_node)
                current_node = path_tracker[current_node]
            path.append(start)


            path.reverse()
            return path, actual_cost[goal]

        #neighbors node
        for neighbor, cost in adj_list[current_node]:
            new_cost = actual_cost[current_node] + cost
            if neighbor not in actual_cost or new_cost < actual_cost[neighbor]:
                path_tracker[neighbor] = current_node
                actual_cost[neighbor] = new_cost
                f_n = new_cost + heuristics[neighbor]
                heapq.heappush(frontier, (f_n, neighbor))

    return None


start_city = input("Start City: ")
goal_city = input("Destination City: ")
path, total_cost = a_star_search(start_city, goal_city)
print("Path:", path)
print("Total distance:", total_cost)
