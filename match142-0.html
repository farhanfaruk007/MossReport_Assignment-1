import heapq

def astar(start, end, adjacent_gra, heuristic):
    frontier = []
    heapq.heappush(frontier, (0, start))
    parent = {}
    cost = {}
    parent[start] = None
    cost[start] = 0
    
    while frontier:
        temp = heapq.heappop(frontier)[1]

        if temp == end:
            break
        for child in adjacent_gra[temp]:
            new_cost = cost[temp] + adjacent_gra[temp][child]
            if child not in cost or new_cost < cost[child]:
                cost[child] = new_cost
                priority = new_cost + heuristic[child]     


                heapq.heappush(frontier, (priority, child))
                parent[child] = temp
                
    return parent, cost
