from queue import PriorityQueue

def dijks(graph,source,dest):

    visited_nodes =PriorityQueue()
    visited_nodes.put((graph[source]['Heuristic'],source))
    explored_nodes=set()
    cost={source:0}
    estimated_dis={source:graph[source]['Heuristic']}

    path={}
    while not visited_nodes.empty():
        if source==dest:
            return (cost[source],source)


        current_node=visited_nodes.get()[1]


        if current_node==dest:
            backtrack=[dest]


            while current_node in path:
                current_node=path[current_node]
                backtrack.append(current_node)
            backtrack.reverse()
            return (estimated_dis[dest],backtrack)

        explored_nodes.add(current_node[1])



        for neighbor,info in graph[current_node]["Neighbor"].items():
            path_cost=cost[current_node]+info


            if neighbor in explored_nodes:
                continue

            elif neighbor not in cost or path_cost<cost[neighbor]:
                cost[neighbor]=path_cost
                estimated_dis[neighbor]=path_cost+graph[neighbor]['Heuristic']
                visited_nodes.put((estimated_dis[neighbor],neighbor))
                path[neighbor]=current_node
    return

graph = {}
inp_file = open('Input file.txt','r')
lines=inp_file.readlines()

for line in lines:
    info=line.split()
    city, heuristic,  neighbor_info= info[0],int(info[1]), {}

    n_i=2
    while n_i<len(info):
        neighbor_city=info[n_i]
        distance=float(info[n_i+1])
        neighbor_info[neighbor_city]=distance
        n_i+=2

    graph[city]={'Heuristic':heuristic,'Neighbor':neighbor_info}

start_point=input('Starting point: ')
end_point=input('End point: ')
res=dijks(graph,start_point, end_point)

if res==None:
    print('NO PATH FOUND')
else:
    ans='Path: '
    for i in res[1]:
        ans+=str(i)+' -> '
    ans=ans[:-4]+f'\nTotal Distance: {res[0]} km'

    print(ans)
