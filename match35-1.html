import heapq

def get_heuristic_values(file_path):
  heruistic_values = {}
  try:


    with open(file_path, 'r') as file:
      for line in file:
        parts = line.strip().split()
        if len(parts) >= 2:
          heruistic_values[parts[0]] = int(parts[1])

  except FileNotFoundError:
    print(f"File '{file_path}' not found.")
  return heruistic_values


def get_adjacency_graph(file_path):
  adjacency_graph={}
  try:
      with open(file_path, 'r') as file:
        for line in file:
          parts = line.strip().split()
          node = parts[0]
          graaph_edges={}

          for index in range(2, len(parts),2):
              neighbor=parts[index]
              cost=int(parts[index+1])
              graaph_edges[neighbor]=cost
          adjacency_graph[node]=graaph_edges
  except FileNotFoundError:
    print(f"File '{file_path}' not found.")


  return adjacency_graph


def a_star_algorithm(start, goal,graph,heuristic):
  priority_queue=[]
  heapq.heappush(priority_queue,(0,start))
  parent_map={start:None}
  cost_map={start: 0}

  while priority_queue:
    current_node=heapq.heappop(priority_queue)[1]

    if current_node==goal:
      break

    for neighbor,weight in graph.get(current_node,{}).items():
      new_cost= cost_map[current_node]+weight
      if neighbor not in cost_map or new_cost<cost_map[neighbor]:
        cost_map[neighbor]=new_cost
        priority=new_cost+heuristic.get(neighbor, float('inf'))
        heapq.heappush(priority_queue,(priority,neighbor))
        parent_map[neighbor]=current_node
  return parent_map,cost_map


def print_path(parent_map,cost_map,start,goal):
  path=[]
  current_node=goal

  while current_node is not None:
    path.append(current_node)
    current_node=parent_map.get(current_node)
  path.reverse()
  if not path or path[0]!=start:
    print("No path found")
  else:
    print("Path:","-->".join(path))
    print(f"Total distance: ",{cost_map[goal]},"km")


def main():
  input_file_path="Input_file.txt.txt"


  heuristic_values=get_heuristic_values(input_file_path)
  adjacency_graph=get_adjacency_graph(input_file_path)

  start_node=input("Enter the starting node: ").strip()
  goal_node=input("Enter the goal node: ").strip()

  if  start_node not in heuristic_values or goal_node not in heuristic_values:
    print("starting or goal node not found in heuristic values.")
    return

  parent_map,cost_map=a_star_algorithm(start_node,goal_node,adjacency_graph,heuristic_values)
  print_path(parent_map,cost_map,start_node,goal_node)

if __name__=="__main__":
    main()
