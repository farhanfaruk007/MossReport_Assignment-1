import heapq
input_txt = open("input.txt", "r")

def parse_heuristics(input_txt):
    city_heuristics = {}
    neighbor_heuristics = {}

    for x in input_txt:
        city_data = x.strip().split()
        city_name = city_data[0]
        city_heuristics[city_name] = int(city_data[1])
        
        neighbors = {}
        for i in range(2, len(city_data), 2):
            neighbor_city = city_data[i]


            neighbor_cost = int(city_data[i + 1])
            neighbors[neighbor_city] = neighbor_cost

        neighbor_heuristics[city_name] = neighbors

    return city_heuristics, neighbor_heuristics

city_heuristics, neighbor_heuristics = parse_heuristics(input_txt)


input_txt.close()

def a_star(start, goal, neighbors, heuristics):
    visited = []
    heapq.heappush(visited, (0, start))
    path = {start: None}
    cost = {start: 0}

    while visited:
        current_city = heapq.heappop(visited)[1]

        if current_city == goal:
            break

        for neighbor in neighbors.get(current_city, {}):
            new_cost = cost[current_city] + neighbors[current_city][neighbor]
            if neighbor not in cost or new_cost < cost[neighbor]:
                cost[neighbor] = new_cost
                priority = new_cost + heuristics[neighbor]


                heapq.heappush(visited, (priority, neighbor))
                path[neighbor] = current_city

    return path, cost

def build_output(path, cost, start, goal):
    route = []
    current = goal
    while current != start:
        route.append(current)
        current = path[current]
    route.append(start)

    if route:
        route.reverse()
        print("Path:", " --> ".join(route))
    else:
        print("No path found.")
    
    print(f'Total distance: {cost[goal]} km')

start = input("Enter the start city: ")
goal = input("Enter the goal city: ")

if start not in city_heuristics or goal not in city_heuristics:
    print("Start or goal does not exist.")
else:
    path, cost = a_star(start, goal, neighbor_heuristics, city_heuristics)
    build_output(path, cost, start, goal)
