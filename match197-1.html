import heapq

def read_file_to_parse(file_path):
    fun_values = {}
    play_area = {}

    with open(file_path, 'r') as file:
        for line in file:
            parts = line.strip().split()
            if len(parts) < 2:
                continue
            if '...' in parts:
                continue


            spot = parts[0]
            fun_value = int(parts[1])
            fun_values[spot] = fun_value
            roads = [(parts[i], int(parts[i + 1])) for i in range(2, len(parts), 2)]
            play_area[spot] = roads

    return fun_values, play_area

def a_star_fun_trip(play_area, fun_values, start, finish):
    todo_list = []
    heapq.heappush(todo_list, (0 + fun_values[start], 0, start, []))
    visited_spots = set()

    while todo_list:
        _, step_cost, current_spot, path_so_far = heapq.heappop(todo_list)

        if current_spot in visited_spots:
            continue

        path_so_far = path_so_far + [current_spot]
        visited_spots.add(current_spot)

        if current_spot == finish:
            return path_so_far, step_cost

        for next_spot, travel_cost in play_area.get(current_spot, []):
            if next_spot not in visited_spots:
                new_step_cost = step_cost + travel_cost
                if next_spot in fun_values:
                    heapq.heappush(todo_list, (new_step_cost + fun_values[next_spot], new_step_cost, next_spot, path_so_far))
                else:
                    print(f"Oops! Skipping {next_spot} as it has no fun value.")
    return None, float('inf')



def main():
    file_path = '/content/sample_data/Lab01.txt'
    fun_values, play_area = read_file_to_parse(file_path)

    start = input("Where do you want to start? ")
    finish = input("Where do you want to go? ")

    if start not in fun_values or finish not in fun_values:
        print("Oops! Either the start or the destination is not valid.")
        return

    path, total_distance = a_star_fun_trip(play_area, fun_values, start, finish)

    if path is None:
        print("Sorry, no path found!")
    else:
        print(f"Here's your path: {' -> '.join(path)}")
        print(f"Total travel distance: {total_distance} km")

if __name__ == "__main__":
    main()
