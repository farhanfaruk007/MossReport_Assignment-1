import heapq

def read_graph(filename):
    graph = {}
    heuristics = {}

    with open(filename, 'r') as file:
        for line in file:
            parts = line.split()
            city = parts[0]
            heuristic = int(parts[1])
            heuristics[city] = heuristic
            neighbors = {}

            for i in range(2, len(parts), 2):
                neighbor = parts[i]
                distance = int(parts[i + 1])
                neighbors[neighbor] = distance



            graph[city] = neighbors

    return graph, heuristics

def a_star_search(graph, heuristics, start, destination):
    open_set = []
    heapq.heappush(open_set, (0, start))

    came_from = {}
    g_score = {city: float('inf') for city in graph}
    g_score[start] = 0

    f_score = {city: float('inf') for city in graph}
    f_score[start] = heuristics[start]

    while open_set:
        current_f, current = heapq.heappop(open_set)

        if current == destination:
            return reconstruct_path(came_from, current), g_score[destination]

        for neighbor, distance in graph[current].items():
            tentative_g_score = g_score[current] + distance

            if tentative_g_score < g_score[neighbor]:


                came_from[neighbor] = current
                g_score[neighbor] = tentative_g_score
                f_score[neighbor] = g_score[neighbor] + heuristics[neighbor]
                heapq.heappush(open_set, (f_score[neighbor], neighbor))

    return None, float('inf')

def reconstruct_path(came_from, current):
    path = [current]
    while current in came_from:
        current = came_from[current]
        path.append(current)
    path.reverse()
    return path

def main():
    filename = "/content/sample_data/input.txt"
    graph, heuristics = read_graph(filename)

    start = input("Enter the starting city: ")
    destination = input("Enter the destination city: ")



    if start not in graph or destination not in graph:
        print("Invalid start or destination city.")
        return

    path, total_distance = a_star_search(graph, heuristics, start, destination)

    if path:
        print(f"Path: {' -> '.join(path)}")
        print(f"Total distance: {total_distance} km")
    else:
        print(f"No path found from {start} to {destination}.")

if __name__ == "__main__":
    main()
