import heapq
file_loc = "input.txt"

def check_heuristic_values(file_loc):
    heuristic_value = {}
 


    with open(file_loc, "r") as file:
        for i in file:
            elem = i.strip().split()
            if len(elem) >= 2:
                heuristic_value[elem[0]] = int(elem[1])

    return heuristic_value

def check_graph_map(file_loc):
    graph_map = {}
      
    with open(file_loc, "r") as file:
        for i in file:
            elem = i.strip().split()
            if not elem:
                continue
            node_city = elem[0]
            graph_edges = {}

            for i in range(2, len(elem), 2):
                nearby_city = elem[i]
                path_cost = int(elem[i + 1])
                graph_edges[nearby_city] = path_cost
            graph_map[node_city] = graph_edges



    return graph_map

def a_star_Algo(start, goal, graph, heuristic):
    priority_que = []
    heapq.heappush(priority_que, (0, start))
    parent_map = {start: None}
    path_cost_map = {start: 0}

    while priority_que:
        current_node = heapq.heappop(priority_que)[1]

        if current_node == goal:
            break

        for nearby_city, weight in graph.get(current_node, {}).items():
            new_cost = path_cost_map[current_node] + weight
            if nearby_city not in path_cost_map or new_cost < path_cost_map[nearby_city]:
                path_cost_map[nearby_city] = new_cost
                priority = new_cost + heuristic.get(nearby_city, float('inf'))
                heapq.heappush(priority_que, (priority, nearby_city))
                parent_map[nearby_city] = current_node
    return parent_map, path_cost_map

def print_path(parent_map, path_cost_map, start, goal):
    path = []
    current_node = goal
    while current_node is not None:
        path.append(current_node)
        current_node = parent_map.get(current_node)
    path.reverse()

    if not path or path[0] != start:
        print("NO path found")
    else:
        print("path:", "-->".join(path))
        print(f"Total distance: {path_cost_map[goal]} km")




heuristics_values = check_heuristic_values(file_loc)
graph_map = check_graph_map(file_loc)

start_city = input("Enter the start city: ").strip()
destination_city = input("Enter the destination city: ").strip()


if start_city not in heuristics_values or destination_city not in heuristics_values:
    print("Wrong Start city or destination city")
else:
    parent_map, path_cost_map = a_star_Algo(start_city, destination_city, graph_map, heuristics_values)
    print_path(parent_map, path_cost_map, start_city, destination_city)
