import heapq

class Graph:
    def __init__(self):
        self.edges = {}
        self.heuristics = {}

    def add_edge(self, node_1, node_2, dis):
        if node_1 not in self.edges:
            self.edges[node_1] = []

        if node_2 not in self.edges:
            self.edges[node_2] = []

        self.edges[node_1].append((node_2, dis))


        self.edges[node_2].append((node_1, dis))

    def set_heuristicsa(self, node, value):
        self.heuristics[node] = value

    def get_heuristics(self, node):
        return self.heuristics.get(node, float('inf'))

    def astar(self, start, goal):
        if start not in self.edges or goal not in self.edges:
            return 'NO PATH FOUND'

        lst = [(self.get_heuristics(start), 0, start, [])]
        set_close = set()

        while lst:
            _, cost, node, path = heapq.heappop(lst)

            if node == goal:
                return path + [node], cost

            if node in set_close:
                continue

            set_close.add(node)

            for neighbour, distance in self.edges.get(node, []):
                if neighbour in set_close:
                    continue

                new_path = path + [node]
                new_cost = cost + distance
                f_score = new_cost + self.get_heuristics(neighbour)
                heapq.heappush(lst, (f_score, new_cost, neighbour, new_path))
        
        return 'NO PATH FOUND'


def file_read(input_file):
    graph = Graph()

    input_file = open('Input file.txt', 'r')
    for l in input_file:
        temp = l.strip().split()
        if len(temp) <3:
            continue
        node = temp[0]
        heuristic = int(temp[1])
        graph.set_heuristicsa(node, heuristic)



        for j in range(2, len(temp), 2):
            neighbour = temp[j]
            distance = int(temp[j+1])
            graph.add_edge(node, neighbour, distance)

    return graph


def final(file_input):
    graph = file_read(file_input)
    start = input("Enter your Starting Node: ").strip()
    goal = input("Enter your Destination: ").strip()
    path, cost = graph.astar(start, goal)

    if path == 'NO PATH FOUND':
        print('NO PATH FOUND')
    else:
        print(f"Path: {' -> '.join(path)}")
        print(f"Total distance: {cost} km")


if __name__ == '__main__':
    file_input = '/content/Input file.txt'
    final(file_input)
